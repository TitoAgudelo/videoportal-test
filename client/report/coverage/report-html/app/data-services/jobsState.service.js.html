(function () {
  'use strict';
 
  angular
    .module('app.data')
    .factory('jobsStateService', jobsStateService);
 
  jobsStateService.$inject = ['$http', '$q', 'exception', 'logger', 'siteSettings'];
  /* @ngInject */
  function jobsStateService($http, $q, exception, logger, siteSettings) {
 
 
    this.jobMatches = [];
    this.selectedJob = null;
    this.currentJobTab = "Match Analysis";
 
    this.jobSearchDateScope = "This Month";
    this.jobSearchLocation = null;
    this.jobSearchMinSalary = null;
    this.jobSearchMaxSalary = null;
    this.jobSearchSkills = [];
 
    this.isLoadingJobMatches = false;
    this.isLoadingJobDetails = false;
 
    this.getJobMatches = function () {
 
      if (this.useTestData) {
        this.jobMatches = this.test.jobs_matches;
        return;
      }
 
      var that = this;
 
      var searchOptions = {
        MaxResults: 20,
        CurrentPage: 0,
        FiltersData: []
      };
 
      for (var i = 0; i < this.jobFiltersUiData.length; i++) {
        for (var j = 0; j < this.jobFiltersUiData[i].FilterDatas.length; j++) {
          searchOptions.FiltersData.push(this.jobFiltersUiData[i].FilterDatas[j]);
        }
      }
 
      that.jobMatches = [];
      that.selectedJob = null;
      that.isLoadingJobMatches = true;
 
      var promise = $http({
        method: 'POST',
        url: siteSettings.apiBaseUrl + '/jobs/matches',
        data: searchOptions
      });
 
      promise.success(function (data, status) {
 
        that.isLoadingJobMatches = false;
 
        that.jobMatches = data;
 
        //that.getJobDetails(data[0].JobId);
      });
 
      promise.error(function (data) {
        that.isLoadingJobMatches = false;
        // TODO: Nathali: show error message in the UI here
      });
    };
 
    this.addJobSearchSkill = function () {
 
      if (this.newJobSearchSkill != "") {
        this.jobSearchSkills.push(this.newJobSearchSkill);
        this.newJobSearchSkill = "";
      }
    };
 
    this.removeJobSearchSkill = function (skill) {
 
      var indexToRemove = this.jobSearchSkills.indexOf(skill);
 
      if (indexToRemove != -1) {
        this.jobSearchSkills.splice(indexToRemove, 1);
      }
    };
 
    this.getJobDetails = function (jobId) {
 
      mixpanel.track("Clicked a job in the job widget");
 
      if (this.useTestData) {
        this.selectedJob = this.test.jobs;
        return;
      }
 
      var that = this;
 
      var timer = window.setTimeout(function () {
        that.isLoadingJobDetails = true;
        that.selectedJob = null;
      }, 100);
 
      var promise = $http({
        method: 'GET',
        url: siteSettings.apiBaseUrl + '/jobs/' + jobId
      });
 
      promise.success(function (data, status) {
        clearTimeout(timer);
 
        that.selectedJob = data;
        that.isLoadingJobDetails = false;
        location.hash = "/jobs/" + jobId;
      });
 
      promise.error(function (data) {
        that.isLoadingJobDetails = false;
 
        that.selectedJob = null;
        // TODO: Nathali, show an error message indicating we coulnd't get a result from the server
      });
    };
 
    this.getApplyUrl = function (jobId) {
 
      if (jobId) {
        return "/jobs/" + jobId + "/apply";
      }
      else {
        return "";
      }
    };
 
    this.getEnrollUrl = function (templateId) {
      return "/course-recommendation/" + templateId + "/enroll";
    };
 
    this.jobFiltersUiData = [];
 
    ///Accepts either an array of filters to add, or a single filter
    this.addOrUpdateJobFilterData = function (arg) {
 
      var filters = [];
 
      if (Object.prototype.toString.call(arg) === '[object Array]') {
        filters = arg;
      }
      else {
        filters.push(arg);
      }
 
      for (var i = 0; i < filters.length; i++) {
        var filterUiData = filters[i];
 
        // track in separate thread
        setTimeout(function () {
          mixpanel.track("JobFilter", {
            Action: 'add',
            Name: filterUiData.uiid,
          });
        }, 100);
 
        var existingFilterUiData = this.jobFiltersUiData.find(function (el) {
          return el && el.uiid == filterUiData.uiid;
        });
 
        if (existingFilterUiData) {
          existingFilterUiData.Label = filterUiData.Label;
          existingFilterUiData.FilterDatas = filterUiData.FilterDatas;
        } else {
          this.jobFiltersUiData.push(filterUiData);
        }
      }
 
      this.getJobMatches();
    };
 
    this.removeFilterData = function (uiid) {
 
      var existingFilterUiData = this.jobFiltersUiData.find(function (el) {
        return el && el.uiid == uiid;
      });
 
      if (existingFilterUiData) {
        this.jobFiltersUiData.splice(this.jobFiltersUiData.indexOf(existingFilterUiData), 1);
 
        this.getJobMatches();
 
        // track in separate thread
        setTimeout(function () {
          mixpanel.track("JobFilter", {
            Action: 'removed',
            Name: uiid,
          });
        }, 100);
      }
    };
 
    return this;
 
  }
})();
