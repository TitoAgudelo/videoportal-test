(function () {
  'use strict';
 
  angular
    .module('app.data')
    .factory('notificationsStateService', notificationsStateService);
 
  notificationsStateService.$inject = ['$http', '$q', 'exception', 'logger', '$timeout', 'siteSettings'];
 
  /* @ngInject */
  function notificationsStateService($http, $q, exception, logger, $timeout, siteSettings) {
 
    var service = {
      notifications: [],
      getUnreadCount: getUnreadCount,
      loadNotifications: loadNotifications,
      markNotificationAsRead: markNotificationAsRead,
      ignoreNotification: ignoreNotification
    };
 
    return service;
 
    /**gets the total number of unread notifications */
    function getUnreadCount() {
      return service.notifications.filter(function (notification) {
        return !notification.IsRead && !notification.Ignored;
      }).length;
    }
 
    /**loads the notifications from the server and updates the state */
    function loadNotifications() {
      return getNotifications()
        .then(function (data) {
          service.notifications = data;
          return data;
        });
    }
 
    /*get all notifications for the current user from the server */
    function getNotifications() {
 
      return $http.get(siteSettings.apiBaseUrl + '/api/v4/notifications')
        .then(success);
 
      function success(response) {
        return response.data;
      }
    };
 
    /*marks a notification as ignored */
    function ignoreNotification(notificationId) {
 
      return $http.post(siteSettings.apiBaseUrl + '/api/v4/notifications/' + notificationId + '/ignore')
        .then(success);
 
      function success(response) {
        service.notifications.forEach(function (notification) {
          if (notification.NotificationId == notificationId) {
            notification.Ignored = true;
          }
        });
      }
    }
 
    /*marks a notification as read */
    function markNotificationAsRead(notificationId) {
 
      return $http.post(siteSettings.apiBaseUrl + '/api/v4/notifications/' + notificationId + '/mark-read')
        .then(success);
 
      function success(response) {
        service.notifications.forEach(function (notification) {
          if (notification.NotificationId == notificationId) {
            notification.IsRead = true;
          }
        });
      }
    }
 
  }
})();
