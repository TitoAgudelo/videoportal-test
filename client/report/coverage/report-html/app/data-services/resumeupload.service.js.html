(function () {
  'use strict';
 
  angular
    .module('app.data')
    .factory('resumeUploadService', resumeUploadService);
 
  resumeUploadService.$inject = ['$http', '$q', 'siteSettings', 'Upload'];
  /* @ngInject */
  function resumeUploadService($http, $q, siteSettings, Upload) {
 
    var self = this;
 
    var service = {
      getCurrentResumeUploadOperationId: getCurrentResumeUploadOperationId,
      getResumeStatusByOperationId: getResumeStatusByOperationId,
      uploadResume: uploadResume
    };
 
    return service;
 
    /**gets the status of a resume uplaod operation given its id */
    function getResumeStatusByOperationId(operationId) {
 
      return $http.get(siteSettings.apiBaseUrl + '/api/v4/resume-upload-operations/' + operationId + '/status')
        .then(success);
 
      function success(response) {
        return response.data;
      }
    }
 
    /**gets the id of the current resume upload operation */
    function getCurrentResumeUploadOperationId() {
      return $http.get(siteSettings.apiBaseUrl + '/api/v4/users/me/resume-upload-operation')
        .then(success);
 
      function success(response) {
        return response.data;
      }
    }
 
    /**uploads the specified resume file */
    function uploadResume(resumeFile) {
      return Upload.upload({
        method: 'PUT',
        url: siteSettings.apiBaseUrl + '/api/v4/resume-upload-operations',
        data: {
          file: resumeFile
        }
      });
    }
  }
})();
