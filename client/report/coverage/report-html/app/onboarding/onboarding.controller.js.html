(function () {
	'use strict';
 
	angular
		.module('app.onboarding')
		.controller('OnboardingController', OnboardingController);
 
	OnboardingController.$inject = ['$q', '$scope', '$rootScope', '$state', 'authService', 'profileStateService', 'usersService'];
 
	/* @ngInject */
	function OnboardingController($q, $scope, $rootScope, $state, authService, profileStateService, usersService) {
		var vm = this;
		vm.title = 'Welcome to Careerscore!';
 
		var vm = this;
		vm.title = 'Onboarding';
		vm.profileState = profileStateService;
		vm.saveLocation = saveLocation;
		vm.City = "";
		vm.State = "";
		vm.Country = "";
		vm.Goals = {
			FindJob: false,
			SelfAnalysis: false,
			LearnDiscover: false
		};
		vm.finish = finish;
 
		// ----------------------SHOW SLIDE 2 AND HIDE SLIDE 1----------------------
		$scope.slide1 = true;
		$scope.slide2 = false;
		$scope.slide3 = false;
		$scope.slide4 = false;
 
		function finish() {
 
			var promises = [
				saveSettings(),
				saveLocation()
			];
 
			return $q.all(promises)
				.then(function () {
					$state.go('profile');
				});
		}
 
		$scope.nextStep = function (stepNumber) {
			// console.log("Next step firing for step...", stepNumber);
			// hide all slides / steps
			$scope.slide1 = false;
			$scope.slide2 = false;
			$scope.slide3 = false;
			$scope.slide4 = false;
 
			// only show the step that was passed in
			if (stepNumber == 1) {
				$scope.slide1 = true;
			} else if (stepNumber == 2) {
				$scope.slide2 = true;
			} else if (stepNumber == 3) {
				$scope.slide3 = true;
			} else if (stepNumber == 4) {
				$scope.slide4 = true;
			}
		};
 
		function saveSettings() {
 
			var settings = {
				IsOnboardingCompleted: true,
				Goals_FindJob: vm.Goals.FindJob,
				Goals_SelfAnalysis: vm.Goals.SelfAnalysis,
				Goals_LearnDiscover: vm.Goals.LearnDiscover
			};
 
			return usersService.saveUISettings(
				settings).then(success);
 
			function success(response) {
				vm.profileState.currentUser.UISettings = response.data;
			}
 
		}
 
		function saveLocation() {
 
			return usersService.updateUserLocation(
				{
					City: vm.City,
					State: vm.State,
					Country: vm.Country
				}).then(success);
 
			function success(data) {
				vm.profileState.currentUser.Location = data;
			}
 
		}
 
 
  }
})();
