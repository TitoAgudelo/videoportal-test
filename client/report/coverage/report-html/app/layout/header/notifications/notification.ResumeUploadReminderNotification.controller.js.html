(function () {
  'use strict';
 
  angular
    .module('app.layout')
    .controller('ResumeUploadReminderNotificationController', ResumeUploadReminderNotificationController);
 
  ResumeUploadReminderNotificationController.$inject = ['$q', '$rootScope', '$scope', '$timeout', 'usersService', 'profileStateService', 'notificationsStateService', 'resumeUploadService'];
 
  /* @ngInject */
  function ResumeUploadReminderNotificationController($q, $rootScope, $scope, $timeout, usersService, profileStateService, notificationsStateService, resumeUploadService) {
 
    var vm = this;
 
    vm.notification = {};
    vm.init = init;
 
    $scope.resumeFile = null;
 
    activate();
 
    function activate() {
      $scope.$watch('resumeFile', function () {
        uploadFile();
      })
    }
 
 
    function init(notification) {
      vm.notification = notification;
    }
 
    /**uploads the current resume file */
    function uploadFile() {
      if ($scope.resumeFile) {
        resumeUploadService.uploadResume($scope.resumeFile);
      }
    }
 
  }
})();
