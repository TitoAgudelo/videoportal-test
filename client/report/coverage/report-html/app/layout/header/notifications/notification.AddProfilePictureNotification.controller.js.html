(function () {
  'use strict';
 
  angular
    .module('app.layout')
    .controller('AddProfilePictureNotificationController', AddProfilePictureNotificationController);
 
  AddProfilePictureNotificationController.$inject = ['$q', '$rootScope', '$scope', '$timeout', 'usersService', 'profileStateService', 'notificationsStateService', 'Upload', 'siteSettings'];
 
  /* @ngInject */
  function AddProfilePictureNotificationController($q, $rootScope, $scope, $timeout, usersService, profileStateService, notificationsStateService, Upload, siteSettings) {
 
    var vm = this;
 
    vm.profilePic = null;
    vm.uploadPicture = uploadPicture;
    vm.init = init;
    vm.index = -1;
    vm.notificationId = '';
 
    var previousPicture;
 
    activate();
 
    function activate() {
      $scope.$watch('vm.profilePic', function () {
        vm.uploadPicture(vm.profilePic);
      });
    }
 
 
    function init(index, notificationId) {
      vm.index = index;
      vm.notificationId = notificationId;
    }
 
 
    function uploadPicture(file) {
      if (file) {
        //show loading thing      
        previousPicture = profileStateService.currentUser.PictureUrl;
        profileStateService.currentUser.PictureUrl = "/images/ajax-loader.gif";
 
        if (!file.$error) {
          Upload.upload({
            url: siteSettings.apiBaseUrl + '/api/v4/users/me/picture',
            data: {
              file: file
            }
          }).success(function (data, status, headers, config) {
 
            if (data.Success) {
              // $scope.changePicture(data.ImageUrl);
              profileStateService.currentUser.PictureUrl = data.ImageUrl;
              $scope.uploadProfilePictureErrors = [];
 
              // ignore this notification right the way
              notificationsStateService.ignoreNotification(vm.notificationId)
            }
          }).error(function (data, status) {
 
            profileStateService.currentUser.PictureUrl = previousPicture;
 
            if (data.Errors) {
              $scope.uploadProfilePictureErrors = data.Errors;
            }
            else {
              $scope.uploadProfilePictureErrors = { "Error": "Unable to upload image." };
            }
          });
        }
      }
    };
 
  }
})();
