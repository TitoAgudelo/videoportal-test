				mixpanel.track("Jobs Widget Advanced Filters clicked");
			}
 
			if ($scope.showAdvancedSearch) //update jobs before closing the advanced panel
			{
				$scope.jobsState.getJobMatches();
			}
 
			$scope.showAdvancedSearch = $scope.showAdvancedSearch ? false : true;
		};

		// Pie Graph creation
		var pieArea = '#donutGraph';

		var pWidth = angular.element(pieArea).width() - 40;
		var pHeight = angular.element(pieArea).height() - 40;
 
		$scope.$watch("jobsState.selectedJob", function () {
 
			$scope.matchAnalysisGraphsDrawn = false;
			$scope.recommendationGraphsDrawn = false;
 
			if ($scope.jobsState.selectedJob !== null) {
 
				$('#companyList').removeAttr("style");
				$('#tail .panelLayout .job-panel-2').removeAttr("style");
				$('#tail .panelLayout .job-panel-3').removeAttr("style");
				$('#tail .panelLayout .job-panel-3 .tab-content').removeAttr("style");
 
				var have = $scope.jobsState.selectedJob.UserMatchInfo.Have;
				var missing = $scope.jobsState.selectedJob.UserMatchInfo.Missing;
				var weCanHelp = $scope.jobsState.selectedJob.UserMatchInfo.WeCanHelpWith;
 
				if (isNaN(have) || isNaN(missing) || isNaN(weCanHelp)) {
					return;
				}
 
				//convert values from decimal
				have = have * 100;
				missing = missing * 100;
				weCanHelp = weCanHelp * 100;
 
				//-----------------------Donut/Pie Graph---------------------
				//clear any previous graph
				angular.element(pieGraph).html("");
 
				var donutData = [];
				donutData.push({ val: have });
				donutData.push({ val: missing });
				donutData.push({ val: weCanHelp });
 
				var pieDimensions = graphService.pieDimensions(pWidth, pHeight);
				var pieSelector = graphService.pieSelector(pWidth, pHeight, donutData);
				var arcCreation = graphService.arcsCreater(pieSelector, pieDimensions.radius, pieDimensions.donut);
 
				graphService.pieCreator(arcCreation, pieDimensions.arc, pieDimensions.labelr, pieDimensions.color);
 
 
				//-----------------------Bar Graphs---------------------
				if ($scope.jobsState.currentJobTab == "Match Analysis") {
 
					$scope.drawMatchAnalysisGraphs();
				}
				else if ($scope.jobsState.currentJobTab == "Recommendations") {
 
					$scope.drawRecommendationGraphs();
				}
 
				companyListHeight();
			}
		});
 
		$scope.$watch("jobsState.currentJobTab", function () {
			if ($scope.jobsState.selectedJob !== null) {
				if ($scope.jobsState.currentJobTab == "Match Analysis" && !$scope.matchAnalysisGraphsDrawn) {
 
					$scope.drawMatchAnalysisGraphs();
				}
				else if ($scope.jobsState.currentJobTab == "Recommendations" && !$scope.recommendationGraphsDrawn) {
 
					$scope.drawRecommendationGraphs();
				}
			}
		});
 
		$scope.drawMatchAnalysisGraphs = function () {
			$timeout(function () {
				var i;
				for (i in $scope.jobsState.selectedJob.MatchAnalysis.Skills) {
					var idNum = Number(i) + 1;
					var jobBarArea = '.jobBarContainer #skillBar' + idNum;
					var element = angular.element(jobBarArea)[0];
 
					var j1width = angular.element(jobBarArea).width();
					var jobChartRange = graphService.getChartRange(element, j1width);
 
					var userBar = $scope.jobsState.selectedJob.MatchAnalysis.Skills[i].UserProficiency * 100; //convert decimal to percent
					var jobBar = $scope.jobsState.selectedJob.MatchAnalysis.Skills[i].RequiredProficiency * 100; //convert decimal to percent
 
					var jobStats = graphService.getRecommendedInfo(userBar, jobBar);
 
					var jobBarbaseArr = graphService.getTailbarBase;
					jobBarbaseArr = graphService.populateTailBarBase(jobBarbaseArr, jobStats);
					var generatedJobArr = graphService.generateData(jobBarbaseArr, jobChartRange);
					var jobScale = graphService.getScale(generatedJobArr, jobChartRange);
					var barData = graphService.initializeGraph(jobBarArea, generatedJobArr, jobScale, graphService.generalObj(1));
 
				}
				$scope.matchAnalysisGraphsDrawn = true;
 
			}, 100); //a slight delay to ensure elements are in place with set widths
		};
 
		$scope.drawRecommendationGraphs = function () {
			$timeout(function () {
				var i;
				for (i in $scope.jobsState.selectedJob.Recommendations) {
 
					var recommendation = $scope.jobsState.selectedJob.Recommendations[i];
 
					if (recommendation.RecommendationType == "CourseRecommendation") {
 
						for (var j = 0; j < recommendation.SkillsToImprove.length; j++) {
 
							var skillToImprove = recommendation.SkillsToImprove[j];
 
							var idNum = Number(j);
							var jobBarArea = '.jobBarContainer #jobBar-' + recommendation.RecommendationId + "-" + idNum;
							var element = angular.element(jobBarArea)[0];
 
							var j1width = angular.element(jobBarArea).width();
							var jobChartRange = graphService.getChartRange(element, j1width);
 
							var userBar = skillToImprove.UserProficiency * 100; //convert decimal to percent
							var jobBar = skillToImprove.RequiredProficiency * 100; //convert decimal to percent
							var courseBar = skillToImprove.IncreasedProficiency * 100; //convert decimal to percent
 
							var jobStats = graphService.getRecommendedInfo(userBar, courseBar, jobBar);
 
							var jobBarbaseArr = graphService.getTailbarBase;
							jobBarbaseArr = graphService.populateTailBarBase(jobBarbaseArr, jobStats);
							var generatedJobArr = graphService.generateData(jobBarbaseArr, jobChartRange);
							var jobScale = graphService.getScale(generatedJobArr, jobChartRange);
							var barData = graphService.initializeGraph(jobBarArea, generatedJobArr, jobScale, graphService.generalObj());
						}
					}
				}
				$scope.recommendationGraphsDrawn = true;
 
			}, 100); //a slight delay to ensure elements are in place with set widths
		};
 
		$scope.getMatchAnalysisClass = function (skill) {
 
			if (skill.WeHaveRecommendations) {
				return "label-yellow";
			}
 
			if (skill.UserProficiency >= skill.RequiredProficiency) {
				return "label-green";
			}
 
			return "label-red";

					$('#tail .panelLayout .job-panel-3 .tab-content').css('height', $('#tail .panelLayout .job-panel-1').height());
				}

		noUiSlider.create(nonLinearSlider, {
