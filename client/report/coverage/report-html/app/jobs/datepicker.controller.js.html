(function () {
  'use strict';
 
  angular
    .module('app.jobs')
    .controller('DatepickerController', DatepickerController);
 
  DatepickerController.$inject = ['$scope',];
  /* @ngInject */
  function DatepickerController($scope) {
    var vm = this;
 
    var vm = this;
 
    vm.valuationDate = new Date();
    vm.valuationDate.setDate(vm.valuationDate.getDate() - 30);
 
    vm.valuationDatePickerIsOpen = false;
    vm.opens = [];
 
    vm.changeTest = function () {
      $scope.userService.addOrUpdateJobFilterData({
        uiid: "PostedSince",
        Label: "Since " + vm.valuationDate.toDateString(),
        FilterDatas: [
          {
            FilterType: "PostedSince",
            FilterValue: moment(vm.valuationDate).format("YYYY/MM/DD")
          }
        ]
      });
    };
 
    $scope.$watch(function () {
      return vm.valuationDatePickerIsOpen;
    }, function (value) {
      vm.opens.push("valuationDatePickerIsOpen: " + value + " at: " + new Date());
    });
 
    vm.valuationDatePickerOpen = function ($event) {
 
      if ($event) {
        $event.preventDefault();
        $event.stopPropagation(); // This is the magic
      }
      this.valuationDatePickerIsOpen = true;
    };
 
  }
})();
