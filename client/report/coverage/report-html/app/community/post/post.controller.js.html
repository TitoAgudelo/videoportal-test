(function () {
	'use strict';
 
	angular
		.module('app.community.post')
		.controller('PostController', PostController);
 
	PostController.$inject = ['logger', '$scope', 'postService', 'containersService', '$state', 'containerPostService', 'profileStateService', '$uibModal'];
	/* @ngInject */
	function PostController(logger, $scope, postService, containersService, $state, containerPostService, profileStateService, $uibModal) {
		var vm = this;
		vm.post = [];
		vm.request = true;
		vm.endPoint = false;
		vm.title = 'Post';
 
		// instance for get post by id
		vm.getPost = getPost;
		vm.getMomentDate = getMomentDate;
		// get current username
		vm.getCurrentUser = getCurrentUser;
		// instance for create post
		vm.createPost = createPost;
		// update Post
		vm.updatePost = updatePost;
		// instance for open create post
		vm.openCreate = openCreate;
		vm.isOpen = false;
		vm.refresh = refresh;
		// instance for upload images 
		vm.upload = upload;
		vm.urlImage = urlImage;
		vm.removeImage = removeImage;
		// instance toggle to add tags
		vm.openTags = openTags;
		vm.isOpenTag = false;
		vm.newTag = newTag;
		// list playlist
		vm.getPlayList = getPlayList;
		vm.openPlayList = openPlayList;
		vm.isOpenPlayList = false;
		vm.addPlayList = addPlayList;
		vm.default = true;
		vm.selectPost = selectPost;
		// instance for get containers in create post
		vm.getContainers = getContainers;
		// add post
		vm.addPost = addPost;
		// edit my post
		vm.editPost = editPost;
		vm.removeTag = removeTag;
		// following
		vm.follow = follow;
		vm.unfollow = unfollow;
		// form to create
		vm.form = {TextContent: '', AttachedPictures: [], Tags: [], ContainerId: '', Container: {}}
 
		vm.containerId = '';
		vm.init = init;
 
		function openPlayList(index) {
			vm.getPlayList();
		}
 
		function getContainers() {
			vm.playlist = [];
			return containersService.getCommunityContainers().then(function(data) {
				vm.isOpenPlayList = !vm.isOpenPlayList;
				vm.playlist = data;
				return vm.playlist;
			});
		} 
 
		function addPlayList(playlist) {
			vm.selectedPlay = playlist;
			vm.form.ContainerId = playlist.ContainerId;
			vm.form.Container = playlist;
			vm.isOpenPlayList = false;
			vm.default = false;
		}
 
		function openCreate() {
			vm.isOpen = vm.isOpen === false ? true: false;
			vm.refresh();
		}
 
		function refresh() {
			vm.form.AttachedPictures = [];
			vm.form.TextContent = '';
			vm.form.Tags = [];
			vm.form.ContainerId = '';
			vm.selectedPlay = {};
			vm.default = true;
		}
 
		function openTags() {
			vm.isOpenTag = vm.isOpenTag === false ? true: false;
			vm.tag = "";
		}
 
		function newTag() {
			if(vm.tag) {
				vm.form.Tags.push(vm.tag);
				vm.openTags();
			}
		}
 
		function urlImage(images) {
			if(images.length > 0) {
				angular.forEach(images, function(file) {
					 vm.form.AttachedPictures.push(file.$ngfBlobUrl);
				});
				return vm.form.AttachedPictures;
			}
		}
 
		function upload(files, errFile) {
			var length = files.length;
			var errFiles = errFile.length; 
			if(length > 0) {
				if(vm.request && vm.endPoint){
					vm.form.AttachedPictures = vm.urlImage(files);
					return;
				}
				vm.form.AttachedPictures = postService.upload(files).then(function(data) {
					var images = [];
					data.forEach(function(d) {
                    	images.push(d.data.Url);
                	});
					return vm.form.AttachedPictures = images;
				});
			}
			if(errFiles > 0) {
				var msg = 'Error trying to upload images',
				title = 'Error upload image';
				toastr.error(msg, title);
			}
		}
 
		function removeImage(index) {
			if (index > -1) {
			    vm.form.AttachedPictures.splice(index, 1);
			}
		}
 
		function addPost(play, post) {
			if(vm.request) {
				vm.post = postService.addPostCommunity(play, post);
				return;
			}
			return postService.addPostCommunity(play, post).then(function(data) {
				vm.post = data;
				return vm.post;
			});
		}
 
		function editPost(data) {
			var containerPost = data.FolloweePostInformations[0].ContainerInfo;
			vm.currentContainer = containerPost;
			$('#myModalPost').modal('show');
			vm.form = {TextContent: '', AttachedPictures: [], Tags: [], ContainerId: '', PostId: ''};
			if(data && vm.request && containerPost) {
				vm.form.ContainerId = containerPost.ContainerId;
				vm.form.AttachedPictures = data.Post.AttachedPictures;
				vm.form.Tags = data.Post.Tags;
				vm.form.TextContent = data.Post.TextContent;
				vm.form.PostId = data.Post.PostId;
			}
		}
 
		function removeTag(currentTag) {
			var index = vm.form.Tags.indexOf(currentTag);
			if (index > -1) {
			    vm.form.Tags.splice(index, 1);
			}
		}
 
		function getPlayList(postId) {
			return containerPostService.getAvailableContainers(postId).then(function(data) {
				vm.containers = data;
				return vm.containers;
			});
		}
 
		function selectPost(container) {
			if(container) {
				vm.form.ContainerId =  container.ContainerId;
				vm.currentContainer = container;
				$scope.isOpenPlayEdit = false;
			}
		}
 
		function getMomentDate(data) {
			angular.forEach(data, function(post) {
				if(post.Post.PostedOnUtc) {
					post.Post.PostedOnUtc = moment(post.Post.PostedOnUtc).startOf('day').fromNow();
				}  
				if(post.FolloweePostInformations.length > 0){
					angular.forEach(post.FolloweePostInformations, function(followee) {
						followee.FolloweeAddedPostOnUtc = moment(followee.FolloweeAddedPostOnUtc).startOf('day').fromNow();
					});
				}
			});
			return data;
		}
 
		function getPost() {
	        var userId = "12c68cb905d2475dbdc3231a00531c00";
 
			if(vm.request && vm.endPoint){
				vm.post = postService.getCommunityPosts(userId, vm.containerId);
				return;
			}
			return postService.getCommunityPosts().then(function(data) {
				vm.post = vm.getMomentDate(data);
				return vm.post;
			});
		}
 
		function addPost(container, post) {
			if(container.IsPostAlreadyAdded === false){
				var self = this;
				self.postId = post.postId;
				return containerPostService.addPostContainer(container.ContainerId , post.PostId).then(function(data) {
					if(data.statusText === "OK") {
						vm.getPlayList(post.PostId);
						vm.getPost();
					}
				});
			} else {
				var msg = 'please select other container not current container or this container is added',
					title = 'Error Current Container';
				toastr.error(msg, title);
			}
		}
 
		function createPost() {
			//function for create community container
	        vm.create = true;
	        if (vm.form.ContainerId && (vm.form.TextContent || vm.form.Tags.length > 0 || vm.form.AttachedPictures.length > 0)) {
				return postService.createCommunityPost(vm.form).then(function(data) {
					vm.getPost();
					vm.openCreate();
	        	});
			} else {
				var msg = 'please select select a container and content',
					title = 'Error create post';
				toastr.error(msg, title);
			}
		}
 
		function updatePost() {
			if(vm.form) {
				return postService.updateCommunityPost(vm.form).then(function(data) {
					if(data.statusText === 'OK') {
						$('#myModalPost').modal('hide');
						vm.getPost();
					} else {
						var msg = 'Error try to update post',
						title = 'Error update post';
						toastr.error(msg, title);
					}
				});
			}
		}
 
		function follow(id, type) {
			return postService.followPost(id, type).then(function(data) {
				vm.getPost();
			});
 
		}
 
		function unfollow(id, type) {
			return postService.unfollowPost(id, type).then(function(data) {
				vm.getPost();
			});
 
		}
 
		function getCurrentUser() {
	      	return profileStateService.loadCurrentUserDetails().then(function(data) {
	        	vm.username = data.Username;
	        	return vm.username;
	      	});
	    }
 
		function init() {
			vm.containerId = $state.params.containerId;
			vm.currentUserName = $state.params.username;
			vm.getCurrentUser();
			vm.getPost();
			vm.default = true;
		}
 
		vm.init();
  	}
})();
