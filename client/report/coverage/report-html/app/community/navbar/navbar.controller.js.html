(function () {
  'use strict';
 
  angular
    .module('app.layout')
    .controller('NavbarController', NavbarController);
 
  NavbarController.$inject = ['$rootScope', '$scope', '$timeout', 'config', 'logger', '$location', '$state', 'profileStateService', 'routerHelper'];
  /* @ngInject */
  function NavbarController($rootScope, $scope, $timeout, config, logger, $location, $state, profileStateService, routerHelper) {
    var vm = this;
    // function for active current location
    var states = routerHelper.getStates();
    vm.getCurrentUser = getCurrentUser;
    vm.currentUser = $rootScope.username;
    vm.isCurrent = isCurrent;
    vm.init = init;
 
    if(vm.currentUser == undefined) {
      vm.init();
    }
 
    function getCurrentUser() {
      return profileStateService.loadCurrentUserDetails().then(function(data) {
        vm.currentUser = data.Username;
        return vm.currentUser;
      });
    }
 
    function init() {
      if(vm.currentUser == null || vm.currentUser == undefined){
        vm.getCurrentUser();
      }
    }
 
    function isCurrent(location) {
      if($state.current.activetab) {
        return $state.current.activetab === location ? 'active' : '';
      }
    }
  }
})();
