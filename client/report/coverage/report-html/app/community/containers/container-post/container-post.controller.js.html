(function () {
	'use strict';
 
	angular
		.module('app.community.containersPost')
		.controller('ContainerPostController', ContainerPostController);
 
	ContainerPostController.$inject = ['logger', '$scope', 'containersService', 'containerPostService', '$state', '$location', 'profileStateService', 'toastr'];
	/* @ngInject */
	function ContainerPostController(logger, $scope, containersService, containerPostService, $state, $location, profileStateService, toastr) {
		var vm = this;
		vm.post = [];
		vm.request = true;
		vm.endPoint = false;
		vm.title = 'Post';
		vm.playlist = {};
		vm.hasUser = hasUser;
		// instance for get post by id
		vm.getPost = getPost;
		vm.getMomentDate = getMomentDate;
		// instance for get post by username
		vm.getPostByUsername = getPostByUsername;
		// instance for get current container
		vm.getCurrentContainer = getCurrentContainer;
		// instance for create post
		vm.createPost = createPost;
		// instance for open create post
		vm.openCreate = openCreate;
		vm.isOpen = false;
		vm.refresh = refresh;
		// instance for upload images 
		vm.upload = upload;
		vm.urlImage = urlImage;
		vm.removeImage = removeImage;
		// instance toggle to add tags
		vm.openTags = openTags;
		vm.isOpenTag = false;
		vm.newTag = newTag;
		vm.removeTag = removeTag;
		// list playlist
		$scope.getPlayList = getPlayList;
		vm.openPlayList = openPlayList;
		vm.isOpenPlayList = false;
		vm.addPlayList = addPlayList;
		vm.default = true;
		vm.addPost = addPost;
		// inatnce for edit
		vm.editPost = editPost;
		vm.updatePost = updatePost;
		vm.deletePost = deletePost;
		// get current user
	    vm.getCurrentUser = getCurrentUser;
	    vm.username = '';
		// form to create
		vm.form = {TextContent: '', AttachedPictures: [], Tags: [], ContainerId: '', PostId: ''};
 
		vm.containerId = '';
		vm.init = init;
 
		function openPlayList() {
			vm.isOpenPlayList = vm.isOpenPlayList === false ? true: false;
		}
 
		function addPlayList(playlist) {
			vm.selectedPlay = playlist;
			vm.form.ContainerId = playlist.ContainerId;
			vm.isOpenPlayList = false;
			vm.default = false;
		}
 
		function openCreate() {
			vm.isOpen = vm.isOpen === false ? true: false;
			vm.refresh();
		}
 
		function refresh() {
			vm.form.AttachedPictures = [];
			vm.form.TextContent = '';
			vm.form.Tags = [];
			vm.form.ContainerId = '';
			vm.selectedPlay = {};
			vm.default = true;
		}
 
		function openTags() {
			vm.isOpenTag = vm.isOpenTag === false ? true: false;
			vm.tag = "";
		}
 
		function newTag() {
			if(vm.tag) {
				vm.form.Tags.push(vm.tag);
				vm.openTags();
			}
		}
 
		function removeTag(currentTag) {
			var index = vm.form.Tags.indexOf(currentTag);
			if (index > -1) {
			    vm.form.Tags.splice(index, 1);
			}
		}
 
		function urlImage(images) {
			if(images.length > 0) {
				angular.forEach(images, function(file) {
					 vm.form.AttachedPictures.push(file.$ngfBlobUrl);
				});
				return vm.form.AttachedPictures;
			}
		}
 
		function upload(files, errFile) {
			var length = files.length;
			var errFiles = errFile.length;
			if(length > 0) {
				if(vm.request && vm.endPoint){
					vm.form.AttachedPictures = vm.urlImage(files);
					return;
				}
				vm.form.AttachedPictures = containerPostService.upload(files).then(function(data) {
					var images = [];
					data.forEach(function(d) {
                    	images.push(d.data.Url);
                	});
					return vm.form.AttachedPictures = images;
				});
			}
			if(errFiles > 0) {
				var msg = 'Error trying to upload images',
				title = 'Error upload image';
				toastr.error(msg, title);
			}
		}
 
		function removeImage(index) {
			if (index > -1) {
			    vm.form.AttachedPictures.splice(index, 1);
			}
		}
 
		function getMomentDate(data) {
			angular.forEach(data, function(post) {
				if(post.UpdatedOnUtc) {
					post.UpdatedOnUtc = moment(post.UpdatedOnUtc).startOf('day').fromNow();
				}
			});
			return data;
		}
 
		function editPost(data) {
			$('#myModalPost').modal('show');
			vm.form = {TextContent: '', AttachedPictures: [], Tags: [], ContainerId: '', PostId: ''};
			if(data && vm.request) {
				vm.form.ContainerId = vm.currentContainer.ContainerId;
				vm.form.AttachedPictures = data.AttachedPictures;
				vm.form.Tags = data.Tags;
				vm.form.TextContent = data.TextContent;
				vm.form.PostId = data.PostId;
			}
		}
 
		function getCurrentUser() {
	      	return profileStateService.loadCurrentUserDetails().then(function(data) {
	        	vm.username = data.Username;
	        	return vm.username;
	      	});
	    }
 
		function updatePost() {
			if(vm.request && vm.endPoint){
				if (vm.form.TextContent || vm.form.AttachedPictures.length > 0 
					|| vm.form.Tags.length > 0 || vm.form.ContainerId){
					vm.form.ContainerId = vm.containerId;
					vm.post = containerPostService.updateCommunityPost(vm.form);
					$('#myModalPost').modal('hide');
					vm.isOpen = false;
					return;
				}
			}
			if (vm.form) {
				return containerPostService.updateCommunityPost(vm.form).then(function(data) {
					if(data.statusText === "OK") { 
						$('#myModalPost').modal('hide');
						vm.getPost();
					}
	        	});
			}
		}
 
		function deletePost(postId) {
			if (postId) {
				if(vm.request && vm.endPoint){
					var postId = vm.form.PostId;
					vm.post = containerPostService.deleteCommunityPost(postId);
					$('#myModalPost').modal('hide');
					vm.isOpen = false;
					return;
					
				}
				return containerPostService.deleteCommunityPost(vm.containerId, postId).then(function(data) {
					if(data.statusText === "OK") {
						$('#myModalPost').modal('hide');
						vm.getPost();
					}
	        	});
			}
		}
 
		function getPost() {
	        var userId = "12c68cb905d2475dbdc3231a00531c00";
	        vm.getCurrentUser();
			if(vm.request && vm.endPoint){
				vm.post = containerPostService.getCommunityPosts(userId, vm.containerId);
				vm.playlist = vm.post[0].ContainerInfo;
				return;
			}
			return containerPostService.getCommunityPosts(userId ,vm.containerId).then(function(data) {
				vm.post = vm.getMomentDate(data);
				return vm.post;
			});
		}
 
		function getCurrentContainer(username) {
			return containersService.getCommunityContainer(username, vm.containerId).then(function(data) {
				vm.currentContainer = data;
				return vm.currentContainer;
			});
		}
 
		function getPlayList(postId) {
			return containerPostService.getAvailableContainers(postId).then(function(data) {
				vm.containers = data;
				return vm.containers;
			});
		}
 
		function addPost(container, postId) {
			if(container.ContainerId !== vm.containerId && container.IsPostAlreadyAdded === false){
				return containerPostService.addPostContainer(container.ContainerId , postId).then(function(data) {
					if(data.statusText === "OK") {
						$scope.getPlayList(postId);
					}
				});
			} else {
				var msg = 'please select other container not current container or this container is added',
					title = 'Error Current Container';
				toastr.error(msg, title);
			}
		}
 
		function createPost() {
			//function for create community container
	        vm.create = true;
			if (vm.form.TextContent || vm.form.AttachedPictures.length > 0 
					|| vm.form.Tags.length > 0 && vm.containerId) {
				if(vm.request && vm.endPoint){
					vm.form.ContainerId = vm.containerId;
					vm.post = containerPostService.createCommunityPost(vm.form, vm.playlist);
					vm.isOpen = false;
					return;
				}
				return containerPostService.createCommunityPost(vm.form, vm.currentContainer).then(function(data) {
					if(data) {
						vm.getPost();
						vm.openCreate();
					}
	        	});
			}
		}
 
		function getPostByUsername(username, containerId) {
			if(username) {
	    		return containerPostService.getPostByUsername(username, containerId).then(function(data) {
					vm.post = vm.getMomentDate(data);
					return vm.post;
				});
	    	}
		}
 
		function hasUser() {
	    	if(vm.currentUserName && vm.username) {
	    		if(vm.currentUserName === vm.username) {
	    			vm.canEdit = true;
	    			vm.getPost();
	    			vm.getCurrentContainer('me');
	    		} else {
	    			vm.canEdit = false;
	    			vm.getPostByUsername(vm.currentUserName, vm.containerId);
	    			vm.getCurrentContainer(vm.currentUserName);
	    		}
	    	}
	    }
 
		function init() {
			vm.containerId = $state.params.containerId;
			vm.currentUserName = $state.params.username;
			vm.getCurrentUser().then(function(data) {
				vm.hasUser();
			});
		}
		vm.init();
  	}
})();
