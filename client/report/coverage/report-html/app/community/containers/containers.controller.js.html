(function () {
	'use strict';
 
	angular
		.module('app.community.containers')
		.controller('ContainersController', ContainersController);
 
	ContainersController.$inject = ['logger', 'containersService', '$scope', '$state', '$location',  'profileStateService'];
	/* @ngInject */
	function ContainersController(logger, containersService, $scope, $state,  $location, profileStateService) {
		var vm = this;
		vm.title = 'Containers';
		vm.init = init;
	    vm.containers = [];
	    // instance for get community container by id
	    vm.getContainer = getContainer;
	    // instance for create community container fn
	    vm.createContainer = createContainer;
	    // instance for delete community container by id
	    vm.deleteContainer = deleteContainer;
	    // instance for update community container by id
	    vm.updateContainer = updateContainer;
	    // instance for reset modal
	    vm.createModal = createModal;
	    // instance for function change color picker
	    vm.changeColor = changeColor;
	    // instance for change bg color of list container
	    vm.setColor = setColor;
	    // button create
	    vm.create = true;
	    // model form for create
	    vm.form = { ContainerName: '', ContainerDescription: '', ContainerColor: '', ContainerPicture: '' }
		// instance for list cotainers
	    vm.listContainers = listContainers;
	    // instance for get containers by username
	    vm.getContainersUser = getContainersUser;
	    // get current user
	    vm.getCurrentUser = getCurrentUser;
	    vm.username = '';
		// instance for toggle function share
		vm.upload = upload;
		// instance function go to post
		vm.goToPost = goToPost;
		$scope.collapsed = false;
		// validate user for public or private content
		vm.hasUser = hasUser;
		// share function
		vm.share = share;
		// request while backend response http request
		vm.request =  true;
		vm.endpoint = false;
 
	    $scope.$watch('vm.containers', function (n, o) {
	        //vm.containers = n;
	        if (n || n == null) {
	            vm.containers = n;
	        }
	    });
 
		function upload(file, errFile) {
			if(file) {
				$scope.onImage = false;
				$scope.onImageModel = true;
				return containersService.upload(file).then(function(data) {
					if(data.statusText === "Created") {
						vm.form.ContainerPicture = data.data.Url;
						return vm.form.ContainerPicture;
					}
				});
			}
		}
 
	    function createModal() {
	        vm.form = {};
			vm.changeColor();
			$('#myModal').modal('show');
	        vm.create = true;
	    }
 
	    // function for change background color on list containers
	    function setColor(color, image) {
	    	if(image) {
	    		var style = {"background-image": "url("+image+") no-repeat center center" };
	    	} else {
				var style = { "background-color": color };
	        	return style;
	    	}
	    }
 
    	// function for change color of create or update icon color picker
	    function changeColor() {
	        var color = vm.form.ContainerColor;
	        if (color || color != '') {
	            vm.colorStyle = { color: color };
	        } else {
	            vm.colorStyle = { color: '#222' };
	        }
	    }
 
	    function getCurrentUser() {
	      	return profileStateService.loadCurrentUserDetails().then(function(data) {
	        	vm.username = data.Username;
	        	return vm.username;
	      	});
	    }
 
	    function goToPost(containerId, containerName) {
	    	$state.go('containersPost', {containerId: containerId, username: vm.currentUserName, containerName: containerName});
	    }
 
	    function listContainers() {
	    	vm.getCurrentUser();
			if(vm.request && vm.endpoint){
				vm.containers = containersService.getCommunityContainers();
				return;
			}
			return containersService.getCommunityContainers().then(function(data) {
				vm.containers = data;
				return vm.containers;
			});
	    }
 
	    function getContainer(containerId) {
	        //function for get community container by id
	        vm.create = false;
	        var userId = "12c68cb905d2475dbdc3231a00531c00";
	        if (userId && containerId) {
				if(vm.request && vm.endpoint){
	        		$('#myModal').modal('show');
					vm.form = containersService.getCommunityContainer(userId, containerId);
					$scope.onImage = false;
					if(vm.form.ContainerPicture) {
						$scope.onImageModel = true;
					} else {
						$scope.onImageModel = false;
					}
					vm.changeColor();
					return;
				}
 
				return containersService.getCommunityContainer(userId, containerId).then(function(data) {
	        		$('#myModal').modal('show');
	            	vm.form = data;
	            	$scope.onImage = false;
					if(vm.form.ContainerPicture) {
						$scope.onImageModel = true;
					} else {
						$scope.onImageModel = false;
					}
					vm.changeColor();
	            	return vm.form;
	        	});
			}
	    }
 
	    function createContainer() {
	        //function for create community container
	        vm.create = true;
			if (vm.form.ContainerName){
				if(vm.request && vm.endpoint){
					vm.containers = containersService.createCommunityContainer(vm.form);
					return;
				}
				containersService.createCommunityContainer(vm.form).then(function(data) {
					if(data.statusText === "OK") {
						vm.listContainers();
					}
	        	});
			}
	    }
 
	    function deleteContainer(containerId) {
	        //function for delete community container by id
	        if (containerId) {
				if(vm.request && vm.endpoint){
					vm.containers = containersService.deleteCommunityContainer(containerId);
					return;
				}
				return containersService.deleteCommunityContainer(containerId).then(function(data) {
					if(data.statusText === "OK") {
						vm.listContainers();
					}
				});
			}
	    }
 
	    function getContainersUser(username) {
	    	if(username) {
	    		return containersService.getContainersByUsername(username).then(function(data) {
					vm.containers = data;
					return vm.containers;
				});
	    	}
	    }
 
	    function updateContainer() {
	        //function for update community container by id
	        if (vm.form.ContainerId && vm.form.ContainerName && vm.form.ContainerDescription) {
	        
				if(vm.request && vm.endpoint){
					vm.containers = containersService.updateCommunityContainer(vm.form);
					return;
				}
				return containersService.updateCommunityContainer(vm.form).then(function(data) {
					if(data.statusText === "OK") {
						vm.listContainers();
					}
				});
			}
	    }
 
	    $scope.shareToggle = function($event) {
	    	$event.stopPropagation();
	    }
 
	    function share($event) {
	    	$event.stopPropagation();
	    }
 
	    function hasUser() {
	    	if(vm.currentUserName && vm.username) {
	    		if(vm.currentUserName === vm.username) {
	    			vm.canEdit = true;
	    			vm.listContainers();
	    		} else {
	    			vm.canEdit = false;
	    			vm.userBy = vm.currentUserName;
	    			vm.getContainersUser(vm.currentUserName);
	    		}
	    	}
	    }
 
	    function init() {
			vm.currentUserName = $state.params.username;
			vm.userBy = '';
			vm.getCurrentUser().then(function(data) {
				vm.hasUser();
			});
		}
 
		vm.init();
  }
})();
