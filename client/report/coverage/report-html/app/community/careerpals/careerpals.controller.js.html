(function () {
	'use strict';
 
	angular
		.module('app.community.career')
		.controller('CareerpalsController', CareerpalsController);
 
	CareerpalsController.$inject = ['logger', 'careerpalServices', '$scope'];
	/* @ngInject */
	function CareerpalsController(logger, careerpalServices, $scope) {
		var vm = this;
		vm.title = 'Careerpals';
		vm.init = init;
		vm.useTest = true;
		vm.filterActive = false;
		// instance for filter
		vm.filter = filter;
		// instance for get user me
		vm.getUserMe = getUserMe;
		vm.getFullUserMe = getFullUserMe;
		vm.getUsername = getUsername;
		vm.getFullUsername = getFullUsername;
		//instance for get followers
		vm.getFollowers =  getFollowers;
		vm.getFollowing = getFollowing;
		// get data
		vm.getData =  getData;
		vm.deleteDuplicates = deleteDuplicates;
		// set Pie highchart data
		vm.setPieData = setPieData;
		vm.getTags = getTags;
 
		function getData(arrayData) {
			angular.forEach(arrayData, function(value, key) {
				vm.follows.push(value);
			});
		}
 
		function deleteDuplicates(newArray) {
		    var cleaned = [];
		    newArray.forEach(function(itm) {
		        var unique = true;
		        cleaned.forEach(function(itm2) {
		            if (itm.Username === itm2.Username) { 
		            	unique = false;
		        	}
		        });
		        if (unique) {
		        	cleaned.push(itm);
		        }
		    });
		    return cleaned;
	    }
 
		function getUserMe() {
			vm.getFullUserMe();
			return careerpalServices.getUserMe().then(function(data) {
				vm.currentUser = data;
				vm.me = data;
				return
			});
		}
 
		function getFullUserMe() {
			return careerpalServices.getFullUserMe().then(function(data) {
				vm.currentUserFull = data;
				vm.meFull = data;
				return
			});
		}
 
		function getUsername(username) {
			vm.getFullUsername(username);
			return careerpalServices.getUsername(username).then(function(data) {
				vm.currentUser = data;
				return
			});
		}
 
		function getFullUsername(username) {
			return careerpalServices.getFullUsername(username).then(function(data) {
				vm.currentUserFull = data;
				return
			});
		}
 
		function filter(model, type) {
			vm.follows = [];
			vm.both = false;
			if(vm.fFollowing && vm.fFollowers) {
				vm.both = true;
			}
			if(model && type === 0 && vm.both === false) {
				vm.getFollowers();
			} else if(model && type === 1 && vm.both === false) {
				vm.getFollowing();
			} else if(model && vm.both) {
				vm.getFollowers().then(function(data) {
					vm.getData(data);
					vm.getFollowing().then(function(data) {
						vm.getData(data);
						if(vm.follows.length > 0) {
							vm.followers = vm.deleteDuplicates(vm.follows);
						} 
					});
				});	
			}
		}
 
		function getFollowers() {
			vm.filterActive = false;
			return careerpalServices.getFollowers().then(function(data) {
				vm.followers = data;
				return data;
			});
		}
 
		function getFollowing() {
			vm.filterActive = false;
			return careerpalServices.getFollowing().then(function(data) {
				vm.followings = data;
				return data;
			});
		}
 
		function setPieData(data) {
			$scope.pieData = [];
			var arry = [];
			angular.forEach(data, function(value, key) {
				var myObj = {};
				myObj.name = value.Tag;
				myObj.y = value.Count;
		        arry.push(myObj);
			});
		    $scope.pieData = arry;
		}
 
		function getTags() {
			return careerpalServices.getTags().then(function(data) {
				vm.setPieData(data);
			});
		}
 
		function init() {
			vm.getFollowers();
			vm.getUserMe();
			vm.getTags()
		}
 
		vm.init();
  	}
})();
