(function () {
  'use strict';
 
  angular
    .module('app.profile')
    .controller('getStartedController', getStartedController);
 
  getStartedController.$inject = ['logger', '$q', '$scope', '$rootScope', '$timeout', 'resumeUploadService'];
  /* @ngInject */
  function getStartedController(logger, $q, $scope, $rootScope, $timeout, resumeUploadService) {
 
    var vm = this;
    vm.upload = upload;
    vm.status = 'undetermined'; // undetermined, uploading, pending, done, failed, 
    vm.currentResumeUploadOperationId = '';
    vm.ready = false;
 
    activate();
 
    function activate() {
      // load current resume operation id and start getting status
      var activationPromises = [
        loadCurrentResumeUploadOperationId()
      ];
 
      return $q.all(activationPromises);
    }
 
    $scope.$watch('resumeFile', function () {
      vm.upload($scope.resumeFile);
    });
 
 
    var onTimeout = function () {
      loadResumeUploadOperationStatus();
      timer = $timeout(onTimeout, 2000);
    };
    var timer = $timeout(onTimeout, 2000);
 
    $scope.$on("$destroy", function () {
      if (timer) {
        $timeout.cancel(timer);
      }
    });
 
 
    /**gets the current resume upload operation id from the api */
    function loadCurrentResumeUploadOperationId() {
 
      return resumeUploadService.getCurrentResumeUploadOperationId()
        .then(function (data) {
          vm.currentResumeUploadOperationId = data;
        });
    }
 
    /**gets the status of the current upload operationid and sets the status */
    function loadResumeUploadOperationStatus() {
      return resumeUploadService.getResumeStatusByOperationId(vm.currentResumeUploadOperationId)
        .then(function (data) {
          vm.status = data.Status.toLowerCase();
          vm.ready = true;
 
          if (vm.status == 'failed') {
            $timeout.cancel(timer);
          }
        });
    }
 
    /**uploads a file provided  */
    function upload(file) {
 
      if (file) {
 
        mixpanel.track("Profile - Upload Resume");
 
        vm.status = 'uploading';
 
        resumeUploadService.uploadResume(file)
          .then(resumeUploadSuccess);
      }
    };
 
    function resumeUploadSuccess(response) {
      vm.status = 'pending';
      vm.currentResumeUploadOperationId = response.data;
      $rootScope.uploadOperationId = response.data;
      $rootScope.$broadcast('Notifications', 'Update');
    }
 
  }
})();
