(function () {
  'use strict';
 
  angular
    .module('app.profile')
    .controller('ProfileEducationController', ProfileEducationController);
 
  ProfileEducationController.$inject = ['$q', '$scope'];
 
  /* @ngInject */
  function ProfileEducationController($q, $scope) {
 
    var vm = this;
    
    /**inidcates if the vm is in choosing mode (the buttons for choosing the education type: bootcamp, degree or course) is open */
    vm.inChoosingMode = false;
    vm.enterChooseMode = enterChooseMode;
    vm.exitChooseMode = exitChooseMode;
    
    /**indicates if the vm is in AddMode, it means if there is an open window 
     * for adding a new event regardless of its type */
    vm.inAddMode = false;
    vm.enterAddMode = enterAddMode;
    vm.exitAddMode = exitAddMode;
 
    vm.addingBootcamp = false;
    vm.addingCourse = false;
    vm.addingDegree = false;
 
    activate();
 
    $scope.$on('profile.education.add.exitAddMode', function () {
      vm.exitAddMode();
    });
 
    function activate() {
 
      var activationPromises = [];
 
      return $q.all(activationPromises);
    }
 
 
    function enterChooseMode(){
      vm.inChoosingMode = true;
    }
 
    function exitChooseMode(){
      vm.inChoosingMode = false;
    }
 
    function enterAddMode() {
      vm.inAddMode = true;
      vm.exitChooseMode();
    }
 
    function exitAddMode() {
      vm.inAddMode = false;
      vm.addingBootcamp = false;
      vm.addingCourse = false;
      vm.addingDegree = false;
      vm.exitChooseMode();
    }
  }
})();
