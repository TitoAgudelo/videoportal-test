(function () {
  'use strict';
 
  angular
    .module('app.profile')
    .controller('CourseDisplayController', CourseDisplayController);
 
  CourseDisplayController.$inject = ['$scope'];
 
  /* @ngInject */
  function CourseDisplayController($scope) {
 
    var vm = this;
    vm.inEditMode = false;
    vm.enterEditMode = enterEditMode;
    vm.exitEditMode = exitEditMode;
 
    $scope.$on('profile.degree.edit.exitEditMode', function () {
      vm.exitEditMode();
    });
 
    function enterEditMode() {
      vm.inEditMode = true;
    }
 
    function exitEditMode() {
      vm.inEditMode = false;
    }
  }
})();
