(function () {
  'use strict';
 
  angular
    .module('app.profile')
    .controller('AddBootcampController', AddBootcampController);
 
  AddBootcampController.$inject = ['$q', '$scope', 'profileStateService', 'usersService'];
 
  /* @ngInject */
  function AddBootcampController($q, $scope, profileStateService, usersService) {
 
    var vm = this;
    vm.exitAddMode = exitAddMode;
 
 
    function exitAddMode() {
      $scope.$emit('profile.education.add.exitAddMode');
    }
 
    vm.addingBootcamp = false;
    vm.newBootcamp = {
      Skills: [],
      ProjectLinks: [""]
    };
    vm.addBootcampErrors = [];
 
    vm.newBootcampDateRange = {
      Start: null,
      End: null
    };
 
    vm.newBootcampNewSkill = '';
    vm.addNewBootcampSkill = addNewBootcampSkill;
    vm.removeNewBootcampSkill = removeNewBootcampSkill;
    vm.addNewBootcamp = addNewBootcamp;
 
    $scope.$watch("vm.newBootcampDateRange.Start", function () {
 
      if (vm.newBootcampDateRange.Start != null) {
        vm.newBootcamp.StartMonth = vm.newBootcampDateRange.Start.getMonth() + 1;
        vm.newBootcamp.StartYear = vm.newBootcampDateRange.Start.getFullYear();
      }
    });
 
    $scope.$watch("vm.newBootcampDateRange.End", function () {
 
      if (vm.newBootcampDateRange.End != null) {
        vm.newBootcamp.EndMonth = vm.newBootcampDateRange.End.getMonth() + 1;
        vm.newBootcamp.EndYear = vm.newBootcampDateRange.End.getFullYear();
      }
    });
 
    function addNewBootcampSkill() {
 
      vm.newBootcampEditSkills = false;
      vm.newBootcamp.Skills.push(vm.newBootcampNewSkill);
      vm.newBootcampNewSkill = "";
    };
 
    function removeNewBootcampSkill(indexToRemove) {
 
      vm.newBootcamp.Skills.splice(indexToRemove, 1);
    };
 
 
    function addNewBootcamp() {
 
      return usersService.addBootcamp(vm.newBootcamp)
        .then(success, error);
 
      function success(data) {
 
        profileStateService.currentUser.BootcampEvents.push(data);
 
        vm.exitAddMode();
 
        vm.newBootcamp = {
          Skills: [],
          ProjectLinks: [""]
        };
        vm.addingBootcamp = false;
        vm.newBootcampAddingProjectLinks = false;
        vm.addBootcampErrors = [];
        vm.newBootcampDateRange = {
          Start: null,
          End: null
        };
 
      };
 
      function error(errors) {
        vm.addBootcampErrors = errors;
      }
 
    };
  }
})();
