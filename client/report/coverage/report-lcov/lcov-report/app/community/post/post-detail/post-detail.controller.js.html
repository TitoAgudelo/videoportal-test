(function () {
	'use strict';
 
	angular
		.module('app.community.post')
		.controller('PostDetailController', PostDetailController);
 
	PostDetailController.$inject = ['logger', '$scope', 'postService', '$state', 'profileStateService', 'authService'];
	/* @ngInject */
	function PostDetailController(logger, $scope, postService, $state, profileStateService, authService) {
		var vm = this;
		vm.post = [];
		vm.request = true;
		vm.endPoint = false;
		vm.title = 'Post';
		vm.init = init;
		vm.public = true;
		vm.getCurrentUser = getCurrentUser;
		// instance get single post
		vm.getPost = getPost;
		vm.getMomentDate = getMomentDate;
		// instance for follow and unfollow
		vm.follow = follow;
		vm.unfollow = unfollow;
 
		function getPost() {
			return postService.getCommunityPost(vm.postId).then(function(data) {
				vm.post = vm.getMomentDate(data);
				return vm.post;
			});
		}
 
		function getMomentDate(post) {
			if(post.PostedOnUtc) {
				post.PostedOnUtc = moment(post.PostedOnUtc).startOf('day').fromNow();
			}  
			return post;
		}
 
		function follow(id, type) {
			return postService.followPost(id, type).then(function(data) {
				vm.getPost();
			});
 
		}
 
		function unfollow(id, type) {
			return postService.unfollowPost(id, type).then(function(data) {
				vm.getPost();
			});
 
		}
 
		function getCurrentUser() {
	      	return profileStateService.loadCurrentUserDetails().then(function(data) {
	        	vm.username = data.Username;
	        	return vm.username;
	      	});
	    }
 
		function init() {
			vm.currentUserName = $state.params.username;
			vm.postId = $state.params.postId;
			if(authService.isAuthenticated) {
				vm.getCurrentUser();
			} else {
				vm.public = false;
			}
			vm.getPost();
		}
 
		vm.init();
  	}
})();
