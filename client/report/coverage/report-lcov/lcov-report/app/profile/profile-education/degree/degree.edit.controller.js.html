(function () {
  'use strict';
 
  angular
    .module('app.profile')
    .controller('ProfileDegreeEditController', ProfileDegreeEditController);
 
  ProfileDegreeEditController.$inject = ['logger', '$q', '$rootScope', '$scope', '$timeout', 'Upload', 'profileStateService', 'usersService'];
 
  /* @ngInject */
  function ProfileDegreeEditController(logger, $q, $rootScope, $scope, $timeout, Upload, profileStateService, usersService) {
 
    var vm = this;
 
    vm.degree = {};
    vm.exitEditMode = exitEditMode;
 
    vm.errors = [];
 
    vm.editAddSkill = {
      newSkill: '',
      adding: false
    };
    vm.addSkill = addSkill;
    vm.removeSkill = removeSkill;
 
    vm.addingMinor = false;
    vm.newMinor = '';
    vm.addMinor = addMinor;
    vm.removeMinor = removeMinor;
 
    vm.editDateRange = {
      Start: null,
      End: null
    };
 
    vm.saveDegree = saveDegree;
    vm.removeDegree = removeDegree;
 
    vm.init = init;
 
 
    /**initializes the controller with the single degree event instance this controller controls */
    function init(degree) {
      vm.degree = degree;
 
      $scope.$watch("vm.editDateRange.Start", function () {
 
        if (vm.editDateRange.Start != null && vm.degree != null) {
          vm.degree.StartMonth = vm.editDateRange.Start.getMonth() + 1;
          vm.degree.StartYear = vm.editDateRange.Start.getFullYear();
        }
      });
 
      $scope.$watch("vm.editDateRange.End", function () {
 
        if (vm.degree.End != null && vm.degree != null) {
          vm.degree.EndMonth = vm.editDateRange.End.getMonth() + 1;
          vm.degree.EndYear = vm.editDateRange.End.getFullYear();
        }
      });
 
      if (vm.degree.StartYear) {
        vm.editDateRange.Start = new Date(vm.degree.StartYear, vm.degree.StartMonth - 1);
      }
      if (vm.degree.EndYear ) {
        vm.editDateRange.End = new Date(vm.degree.EndYear, vm.degree.EndMonth - 1);
      }
    }
 
    /**exits edit mode */
    function exitEditMode() {
      $scope.$emit('profile.degree.edit.exitEditMode');
    }
 
    /**adds a new skill */
    function addSkill() {
      vm.editAddSkill.adding = false;
      vm.degree.Skills.push(vm.editAddSkill.newSkill);
      vm.editAddSkill.newSkill = '';
    }
 
    /**removes a skill */
    function removeSkill(indexToRemove) {
      vm.degree.Skills.splice(indexToRemove, 1);
    }
 
    /**adds a minor */
    function addMinor() {
 
      vm.edittingMinors = false;
      vm.degree.Minors.push(vm.newMinor);
      vm.newMinor = '';
    }
 
    /**remove minor */
    function removeMinor(indexToRemove) {
 
      vm.degree.Minors.splice(indexToRemove, 1);
    }
 
    /**saves the degree */
    function saveDegree() {
 
      usersService.editDegree(vm.degree)
        .then(success, error);
 
      function success() {
 
        vm.exitEditMode();
 
        $timeout(function () {
          usersService.getDegree(vm.degree.ResumeElementId)
            .then(function (data) {
 
              var idx = profileStateService.currentUser.DegreeEvents.indexOf(vm.degree);
              profileStateService.currentUser.DegreeEvents[idx] = data;
 
            });
 
        }, 2500);
      }
 
      function error(errors) {
        vm.errors = errors;
      }
 
    }
 
    /**removes the degree event */
    function removeDegree() {
 
     usersService.removeDegree(vm.degree.ResumeElementId)
        .then(success, error);
 
      function success() {
 
        profileStateService.currentUser.DegreeEvents
          .splice(profileStateService.currentUser.DegreeEvents.indexOf(vm.degree), 1);
 
        vm.exitEditMode();
 
        $timeout(function () {
 
          // load user scores
 
        }, 2500);
      }
 
      function error(errors) {
        vm.errors = errors;
      }
    }
 
  }
})();
