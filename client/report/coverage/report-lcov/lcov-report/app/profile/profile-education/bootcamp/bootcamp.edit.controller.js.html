(function () {
  'use strict';
 
  angular
    .module('app.profile')
    .controller('BootcampEditController', BootcampEditController);
 
  BootcampEditController.$inject = ['logger', '$q', '$rootScope', '$scope', '$timeout', 'profileStateService', 'usersService'];
 
  /* @ngInject */
  function BootcampEditController(logger, $q, $rootScope, $scope, $timeout, profileStateService, usersService) {
 
    var vm = this;
 
    vm.exitEditMode = exitEditMode;
 
    vm.bootcamp = null;
    vm.editBootcampErrors = [];
    vm.editBootcampAddSkill = {
      newSkill: "",
      adding: false
    };
 
    vm.editBootcampDateRange = {
      Start: null,
      End: null
    };
 
    vm.addBootcampSkill = addBootcampSkill;
    vm.removeBootcampSkill = removeBootcampSkill;
    vm.saveBootcamp = saveBootcamp;
    vm.removeBootcamp = removeBootcamp;
 
    vm.init = init;
 
    /**initializes the controller with the single degree event instance this controller controls */
    function init(bootcamp) {
      vm.bootcamp = bootcamp;
 
      $scope.$watch("vm.editBootcampDateRange.Start", function () {
 
        if (vm.editBootcampDateRange.Start != null && vm.bootcamp != null) {
          vm.bootcamp.StartMonth = vm.editBootcampDateRange.Start.getMonth() + 1;
          vm.bootcamp.StartYear = vm.editBootcampDateRange.Start.getFullYear();
        }
      });
 
      $scope.$watch("vm.editBootcampDateRange.End", function () {
 
        if (vm.editBootcampDateRange.End != null && vm.bootcamp != null) {
          vm.bootcamp.EndMonth = vm.editBootcampDateRange.End.getMonth() + 1;
          vm.bootcamp.EndYear = vm.editBootcampDateRange.End.getFullYear();
        }
      });
 
      if (vm.bootcamp.StartYear) {
        vm.editBootcampDateRange.Start = new Date(vm.bootcamp.StartYear, vm.bootcamp.StartMonth - 1);
      }
 
      if (vm.bootcamp.EndYear) {
        vm.editBootcampDateRange.End = new Date(vm.bootcamp.EndYear, vm.bootcamp.EndMonth - 1);
      }
    }
 
    /**exits edit mode */
    function exitEditMode() {
      $scope.$emit('profile.degree.edit.exitEditMode');
    }
 
    function addBootcampSkill() {
 
      vm.editBootcampAddSkill.adding = false;
      vm.bootcamp.Skills.push(vm.editBootcampAddSkill.newSkill);
      vm.editBootcampAddSkill.newSkill = "";
    }
 
    function removeBootcampSkill(indexToRemove) {
 
      vm.bootcamp.Skills.splice(indexToRemove, 1);
    }
 
 
    function saveBootcamp() {
 
      usersService.editBootcamp(vm.bootcamp)
        .then(success, error);
 
      function success() {
 
        vm.exitEditMode();
 
        $timeout(function () {
          usersService.getBootcamp(vm.bootcamp.ResumeElementId)
            .then(function (data) {
 
              var idx = profileStateService.currentUser.BootcampEvents.indexOf(vm.bootcamp);
              profileStateService.currentUser.BootcampEvents[idx] = data;
 
            });
 
        }, 2500);
      }
 
      function error(errors) {
        vm.errors = errors;
      }
    }
 
        /**removes the degree event */
    function removeBootcamp() {
 
     usersService.removeBootcamp(vm.bootcamp.ResumeElementId)
        .then(success, error);
 
      function success() {
 
        profileStateService.currentUser.BootcampEvents
          .splice(profileStateService.currentUser.BootcampEvents.indexOf(vm.bootcamp), 1);
 
        vm.exitEditMode();
 
        $timeout(function () {
 
          // load user scores
 
        }, 2500);
      }
 
      function error(errors) {
        vm.errors = errors;
      }
    }
 
 
  }
})();
