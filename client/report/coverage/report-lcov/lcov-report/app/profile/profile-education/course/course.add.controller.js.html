(function () {
  'use strict';
 
  angular
    .module('app.profile')
    .controller('AddCourseController', AddCourseController);
 
  AddCourseController.$inject = ['$q', '$scope', 'profileStateService', 'usersService'];
 
  /* @ngInject */
  function AddCourseController($q, $scope, profileStateService, usersService) {
 
    var vm = this;
    vm.exitAddMode = exitAddMode;
 
    vm.addingCourse = false;
    vm.newCourse = {
      Skills: [],
      ProjectLinks: [""]
    };
    vm.addCourseErrors = [];
    vm.newCourseDateRange = {
      Start: null,
      End: null
    };
    vm.addNewCourseSkill = addNewCourseSkill;
    vm.removeNewCourseSkill = removeNewCourseSkill;
    vm.addNewCourse = addNewCourse;
 
    $scope.$watch("vm.newCourseDateRange.Start", function () {
 
      if (vm.newCourseDateRange.Start != null) {
        vm.newCourse.StartMonth = vm.newCourseDateRange.Start.getMonth() + 1;
        vm.newCourse.StartYear = vm.newCourseDateRange.Start.getFullYear();
      }
    });
 
    $scope.$watch("vm.newCourseDateRange.End", function () {
 
      if (vm.newCourseDateRange.End != null) {
        vm.newCourse.EndMonth = vm.newCourseDateRange.End.getMonth() + 1;
        vm.newCourse.EndYear = vm.newCourseDateRange.End.getFullYear();
      }
    });
 
    function exitAddMode() {
      $scope.$emit('profile.education.add.exitAddMode');
    }
 
 
    function addNewCourseSkill() {
 
      vm.newCourseEditSkills = false;
      vm.newCourse.Skills.push(vm.newCourseNewSkill);
      vm.newCourseNewSkill = "";
    }
 
    function removeNewCourseSkill(indexToRemove) {
 
      vm.newCourse.Skills.splice(indexToRemove, 1);
    }
 
    function addNewCourse() {
 
      return usersService.addCourse(vm.newCourse)
        .then(success, error);
 
      function success(data) {
 
        profileStateService.currentUser.CourseEvents.push(data);
 
        vm.exitAddMode();
 
        vm.newCourse = {
          Skills: [],
          ProjectLinks: [""]
        };
        vm.addingCourse = false;
        vm.addCourseErrors = [];
        vm.newCourseDateRange = {
          Start: null,
          End: null
        };
 
      };
 
      function error(errors) {
        vm.addBootcampErrors = errors;
      }
     
    }
 
  }
})();
