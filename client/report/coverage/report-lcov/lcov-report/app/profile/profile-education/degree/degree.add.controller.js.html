(function () {
  'use strict';
 
  angular
    .module('app.profile')
    .controller('ProfileDegreeAddController', ProfileDegreeAddController);
 
  ProfileDegreeAddController.$inject = ['$scope', 'profileStateService', 'usersService', 'exception'];
 
  /* @ngInject */
  function ProfileDegreeAddController($scope, profileStateService, usersService, exception) {
 
    var vm = this;
 
    vm.addingDegree = false;
    vm.exitAddMode = exitAddMode;
    vm.newDegree = {
      Skills: [],
      ProjectLinks: [''],
      Minors: [],
      StudyAbroads: []
    };
    vm.newDegreeDateRange = {
      Start: null,
      End: null
    };
    vm.addDegreeErrors = [];
 
    vm.newDegreeNewSkill = '';
    vm.addSkill = addSkill;
    vm.removeSkill = removeSkill;
 
    vm.addProjectLink = addProjectLink;
    vm.removeProjectLink = removeProjectLink;
    vm.addNewDegree = addNewDegree;
 
    vm.edittingMinors = false;
    vm.newMinor = '';
    vm.addMinor = addMinor;
    vm.removeMinor = removeMinor;
 
    vm.addingStudyAbroad = false;
    vm.newStudyAbroad = {
      Skills: [],
      Location: {}
    };
    vm.addStudyAbroadErrors = [];
    vm.newStudyAbroadDateRange = {
      Start: null,
      End: null
    };
    vm.newStudyAbroadNewSkill = '';
    vm.addNewStudyAbroadSkill = addNewStudyAbroadSkill;
    vm.removeNewStudyAbroadSkill = removeNewStudyAbroadSkill;
    vm.addNewStudyAbroad = addNewStudyAbroad;
    vm.removeNewStudyAbroad = removeNewStudyAbroad;
 
 
    $scope.$watch('vm.newDegreeDateRange.Start', function () {
 
      if (vm.newDegreeDateRange.Start != null) {
        vm.newDegree.StartMonth = vm.newDegreeDateRange.Start.getMonth() + 1;
        vm.newDegree.StartYear = vm.newDegreeDateRange.Start.getFullYear();
      }
    });
 
    $scope.$watch('vm.newDegreeDateRange.End', function () {
 
      if (vm.newDegreeDateRange.End != null) {
        vm.newDegree.EndMonth = vm.newDegreeDateRange.End.getMonth() + 1;
        vm.newDegree.EndYear = vm.newDegreeDateRange.End.getFullYear();
      }
    });
 
    $scope.$watch("vm.newStudyAbroadDateRange.Start", function () {
 
      if (vm.newStudyAbroadDateRange.Start != null) {
        vm.newStudyAbroad.StartMonth = vm.newStudyAbroadDateRange.Start.getMonth() + 1;
        vm.newStudyAbroad.StartYear = vm.newStudyAbroadDateRange.Start.getFullYear();
      }
    });
 
    $scope.$watch("vm.newStudyAbroadDateRange.End", function () {
 
      if (vm.newStudyAbroadDateRange.End != null) {
        vm.newStudyAbroad.EndMonth = vm.newStudyAbroadDateRange.End.getMonth() + 1;
        vm.newStudyAbroad.EndYear = vm.newStudyAbroadDateRange.End.getFullYear();
      }
    });
 
    function exitAddMode() {
      $scope.$emit('profile.education.add.exitAddMode');
    }
 
    /**adds a new skill to the new Degree event */
    function addSkill() {
 
      var skillExist = vm.newDegree.Skills.filter(function (s) {
        return s == vm.newDegreeNewSkill;
      }).length > 0;
 
      if (!skillExist) {
        vm.newDegreeEditSkills = false;
        vm.newDegree.Skills.push(vm.newDegreeNewSkill);
      }
      vm.newDegreeNewSkill = '';
    };
 
    /**removes a skill from the new Degree event */
    function removeSkill(indexToRemove) {
 
      vm.newDegree.Skills.splice(indexToRemove, 1);
    };
 
    /**adds a new project link from the new Degree event  */
    function addProjectLink() {
 
      if (vm.newDegree.ProjectLinks.indexOf("") == -1) {
        vm.newDegree.ProjectLinks.push("");
      }
    };
 
    /**removes a project linkn from the new Degree event */
    function removeProjectLink(indexToRemove) {
 
      vm.newDegree.ProjectLinks.splice(indexToRemove, 1);
    }
 
    function addMinor() {
 
      vm.edittingMinors = false;
      vm.newDegree.Minors.push(vm.newMinor);
      vm.newMinor = '';
    }
 
    function removeMinor(indexToRemove) {
 
      vm.newDegree.Minors.splice(indexToRemove, 1);
    }
 
    /**ads the new Degree event */
    function addNewDegree() {
 
      return usersService.addDegree(vm.newDegree)
        .then(success, error);
 
      function success(data) {
 
        profileStateService.currentUser.DegreeEvents.push(data);
 
        vm.exitAddMode();
 
        vm.newDegreeAddingProjectLinks = false;
        vm.newDegree = {
          Skills: [],
          ProjectLinks: [""],
          Minors: [],
          StudyAbroads: []
        };
        vm.addDegreeErrors = [];
        vm.newDegreeDateRange = {
          Start: null,
          End: null
        };
      };
 
      function error(errors) {
        vm.addDegreeErrors = errors;
      }
    }
 
 
 
    function addNewStudyAbroadSkill() {
 
      vm.newStudyAbroadEditSkills = false;
      vm.newStudyAbroad.Skills.push(vm.newStudyAbroadNewSkill);
      vm.newStudyAbroadNewSkill = '';
    }
 
    function removeNewStudyAbroadSkill(indexToRemove) {
 
      vm.newStudyAbroad.Skills.splice(indexToRemove, 1);
    }
 
    function addNewStudyAbroad() {
 
      vm.newDegree.StudyAbroads.push(vm.newStudyAbroad);
 
      vm.newStudyAbroad = {
        Skills: [],
        Location: {}
      };
      vm.addingStudyAbroad = false;
      vm.newStudyAbroadAddingProjectLinks = false;
      vm.addStudyAbroadErrors = [];
      vm.newStudyAbroadDateRange = {
        Start: null,
        End: null
      };
 
    }
 
    function removeNewStudyAbroad(indexToRemove){
      vm.newDegree.StudyAbroads.splice(indexToRemove, 1);
    }
 
  }
})();
