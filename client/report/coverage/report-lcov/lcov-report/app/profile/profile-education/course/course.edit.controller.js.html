(function () {
  'use strict';
 
  angular
    .module('app.profile')
    .controller('CourseEditController', CourseEditController);
 
  CourseEditController.$inject = ['logger', '$q', '$rootScope', '$scope', '$timeout', 'profileStateService', 'usersService'];
 
  /* @ngInject */
  function CourseEditController(logger, $q, $rootScope, $scope, $timeout, profileStateService, usersService) {
 
    var vm = this;
 
    vm.exitEditMode = exitEditMode;
 
    vm.editCourseObj = null;
    vm.editCourseIndex = null;
    vm.editCourseErrors = [];
    vm.editCourseAddSkill = {
      newSkill: "",
      adding: false
    };
    vm.editCourseDateRange = {
      Start: null,
      End: null
    };
    vm.addCourseSkill = addCourseSkill;
    vm.removeCourseSkill = removeCourseSkill;
    vm.saveCourse = saveCourse;
    vm.removeCourse = removeCourse;
 
    vm.init = init;
    
 
    /**initializes the controller with the single degree event instance this controller controls */
    function init(course) {
 
      vm.editCourseObj = course;
 
      $scope.$watch("vm.editCourseDateRange.Start", function () {
 
        if (vm.editCourseDateRange.Start != null && vm.editCourseObj != null) {
          vm.editCourseObj.StartMonth = vm.editCourseDateRange.Start.getMonth() + 1;
          vm.editCourseObj.StartYear = vm.editCourseDateRange.Start.getFullYear();
        }
      });
 
      $scope.$watch("vm.editCourseDateRange.End", function () {
 
        if (vm.editCourseDateRange.End != null && vm.editCourseObj != null) {
          vm.editCourseObj.EndMonth = vm.editCourseDateRange.End.getMonth() + 1;
          vm.editCourseObj.EndYear = vm.editCourseDateRange.End.getFullYear();
        }
      });
 
      if (course.StartYear) {
        vm.editCourseDateRange.Start = new Date(course.StartYear, course.StartMonth - 1);
      }
 
      if (course.EndYear) {
        vm.editCourseDateRange.End = new Date(course.EndYear, course.EndMonth - 1);
      }
    }
 
    /**exits edit mode */
    function exitEditMode() {
      $scope.$emit('profile.degree.edit.exitEditMode');
    }
 
 
    function addCourseSkill() {
 
      vm.editCourseAddSkill.adding = false;
      vm.editCourseObj.Skills.push(vm.editCourseAddSkill.newSkill);
      vm.editCourseAddSkill.newSkill = "";
    }
 
    function removeCourseSkill(indexToRemove) {
 
      vm.editCourseObj.Skills.splice(indexToRemove, 1);
    }
 
 
    function saveCourse() {
 
      usersService.editCourse(vm.editCourseObj)
        .then(success, error);
 
      function success() {
 
        vm.exitEditMode();
 
        $timeout(function () {
          usersService.getCourse(vm.editCourseObj.ResumeElementId)
            .then(function (data) {
 
              var idx = profileStateService.currentUser.CourseEvents.indexOf(vm.editCourseObj);
              profileStateService.currentUser.CourseEvents[idx] = data;
 
            });
 
        }, 2500);
      }
 
      function error(errors) {
        vm.editCourseErrors = errors;
      }
      
    }
 
     function removeCourse() {
 
      usersService.removeCourse(vm.editCourseObj.ResumeElementId)
        .then(success, error);
 
      function success() {
 
        profileStateService.currentUser.CourseEvents
          .splice(profileStateService.currentUser.CourseEvents.indexOf(vm.editCourseObj), 1);
 
        vm.exitEditMode();
 
        $timeout(function () {
 
          // load user scores
 
        }, 2500);
      }
 
      function error(errors) {
        vm.editCourseErrors = errors;
      }
    }
 
  }
})();
