(function () {
	'use strict';
 
	angular
		.module('app.profile')
		.controller('ProfileBucketsController', ProfileBucketsController);
 
	ProfileBucketsController.$inject = ['logger', '$q', '$rootScope', '$scope', '$timeout', 'Upload', 'usersService', 'profileStateService'];
	/* @ngInject */
	function ProfileBucketsController(logger, $q, $rootScope, $scope, $timeout, Upload, usersService, profileStateService) {
 
		var vm = this;
 
		vm.buckets = [];
		vm.newBucket = '';
		vm.bucketDDIsOpen = false;
		vm.getActiveBucket = getActiveBucket;
		vm.hasActiveBucket = hasActiveBucket;
		vm.addBucket = addBucket;
		vm.updateBucket = updateBucket;
 
		activate();
 
		function activate() {
			return getSupportedBuckets();
		}
 
		/**gets the list of buckets that we support */
		function getSupportedBuckets() {
			return usersService.getSupportedBuckets().then(function (data) {
				vm.buckets = data;
				return vm.buckets;
			});
		}
 
		/**determines if there is a value in the activeBucket property */
		function hasActiveBucket() {
			return profileStateService.hasActiveBucket();
		}
 
		/**gets the active bucket */
    function getActiveBucket() {
      return profileStateService.activeBucket;
    }
 
 
		/**adds a new bucket to the list of buckets */
		function addBucket(event) {
			if (event.which === 13) //the user hit the enter key
			{
				vm.buckets.push(vm.newBucket);
				vm.newBucket = '';
				event.preventDefault();
			}
		};
 
		/**updates the current bucket */
		function updateBucket(option) {
 
			// set new bucket as active 
			profileStateService.activeBucket = option;
 
			// send new bucket to server
			usersService.updateUserBucket(option);
 
			// start loading scores until we get a score back
			profileStateService.startLoadingCurrentUserScore();
 
			vm.bucketDDIsOpen = false;
		};
  }
})();
