(function (window) {
	'use strict'; //it wont allow bad syntax
	var angular = window.angular;
 
	angular
		.module('app.profile')
		.controller('ResumeVerificationController', ResumeVerificationController);
 
	ResumeVerificationController.$inject = ['logger', '$scope', 'profileStateService', 'usersService'];
 
	/* @ngInject */
	function ResumeVerificationController(logger, $scope, profileStateService, usersService) {
		var vm = this;
		vm.title = 'ResumeVerification';
		vm.profileState = profileStateService;
		vm.done = done;
 
 
		activate();
 
		function activate() {
 
			// load profile information with rules now.
			profileStateService.loadCurrentUserDetails(true);
 
		}
 
		// on activate, get the userdata with verification rules, and show it here so the user can edit it
		// at the end mark the profile as IsProfileVerified = true
 
 
		// ----------------------SHOW SLIDE 2 AND HIDE SLIDE 1----------------------
		$scope.slide1 = true;
		$scope.slide2 = true;
		$scope.slide3 = false;
		$scope.slide4 = false;
 
		$scope.nextStep = function (stepNumber) {
 
			// should go next if rules are still pending??????
 
			// console.log("Next step firing for step...", stepNumber);
			// hide all slides / steps
			$scope.slide1 = false;
			$scope.slide2 = false;
			$scope.slide3 = false;
			$scope.slide4 = false;
 
			// only show the step that was passed in
			if (stepNumber == 1) {
				$scope.slide1 = true;
			} else if (stepNumber == 2) {
				$scope.slide2 = true;
			} else if (stepNumber == 3) {
				$scope.slide3 = true;
			} else if (stepNumber == 4) {
				$scope.slide4 = true;
			}
		};
 
		function done() {
			$state.go('profile');
		}
	}
})(window);
