(function () {
  'use strict';
 
  angular
    .module('app.profile')
    .controller('EditProfileBasicInfoController', EditProfileBasicInfoController);
 
  EditProfileBasicInfoController.$inject = ['logger', '$q', '$rootScope', '$scope', 'profileStateService', 'usersService', 'exception'];
 
  /* @ngInject */
  function EditProfileBasicInfoController(logger, $q, $rootScope, $scope, profileStateService, usersService, exception) {
 
    var vm = this;
 
    vm.profileState = profileStateService;
    vm.basicInfo = {
      FirstName: '',
      LastName: '',
      Position: '',
      Location: {
        RawLocation: ''
      }
    };
    vm.editBasicInfoErrors = [];
    vm.edittingName = false;
    vm.edittingPosition = false;
    vm.edittingLocation = false;
    vm.updateName = updateName;
    vm.cancelUpdateName = cancelUpdateName;
    vm.updatePosition = updatePosition;
    vm.cancelUpdatePosition = cancelUpdatePosition;
    vm.updateLocation = updateLocation;
    vm.cancelUpdateLocation = cancelUpdateLocation;
 
    activate();
 
    function activate() {
 
      var activationPromises = [];
 
      return $q.all(activationPromises);
    }
 
    /**if editting name, initialize form values */
    $scope.$watch('vm.edittingName', function () {
 
      if (vm.edittingName) {
        syncData();
      }
    });
 
    /**if editting location, initialize form values */
    $scope.$watch('vm.edittingLocation', function () {
 
      if (vm.edittingLocation) {
        syncData();
      }
    });
 
    /**if editting position, initialize form values */
    $scope.$watch('vm.edittingPosition', function () {
 
      if (vm.edittingPosition) {
        syncData();
      }
    });
 
    /**syncs data from currentUser to editInfo model */
    function syncData() {
      vm.basicInfo.FirstName = vm.profileState.currentUser.FirstName;
      vm.basicInfo.LastName = vm.profileState.currentUser.LastName;
      vm.basicInfo.Position = vm.profileState.currentUser.Position;
      vm.basicInfo.Location.RawLocation = vm.profileState.currentUser.Location.RawLocation;
    }
 
    /**updates the name of basic info */
    function updateName() {
 
      return usersService.updateUserBasicInfo(vm.basicInfo)
        .then(success, error);
 
      function success(response) {
        vm.profileState.currentUser.FirstName = vm.basicInfo.FirstName;
        vm.profileState.currentUser.LastName = vm.basicInfo.LastName;
 
        vm.edittingName = false;
        vm.editBasicInfoErrors = [];
      }
    }
 
    /**cancels editting the name */
    function cancelUpdateName() {
      vm.edittingName = false;
      vm.editBasicInfoErrors = [];
    }
 
    /**updates the position of position info */
    function updatePosition() {
 
      return usersService.updateUserBasicInfo(vm.basicInfo)
        .then(success, error);
 
      function success(response) {
        vm.profileState.currentUser.Position = vm.basicInfo.Position;
 
        vm.edittingPosition = false;
        vm.editBasicInfoErrors = [];
      }
    }
 
    /**cancels editting the position */
    function cancelUpdatePosition() {
      vm.edittingPosition = false;
      vm.editBasicInfoErrors = [];
    }
 
    /**updates the location of position info */
    function updateLocation() {
 
      return usersService.updateUserBasicInfo(vm.basicInfo)
        .then(success, error);
 
      function success(response) {
        vm.profileState.currentUser.Location.RawLocation = vm.basicInfo.Location.RawLocation;
 
        vm.edittingLocation = false;
        vm.editBasicInfoErrors = [];
      }
    }
 
    /**cancels editting the location */
    function cancelUpdateLocation() {
      vm.edittingLocation = false;
      vm.editBasicInfoErrors = [];
    }
 
    /** callback for error on saving basic info*/
    function error(response) {
 
      if (response.data.ModelState) {
        vm.editBasicInfoErrors = exception.parseErrors(response.data);
      }
    }
 
 
  }
})();
