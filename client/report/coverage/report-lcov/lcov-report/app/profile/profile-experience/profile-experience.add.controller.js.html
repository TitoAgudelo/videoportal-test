(function () {
  'use strict';
 
  angular
    .module('app.profile')
    .controller('ProfileExperienceAddController', ProfileExperienceAddController);
 
  ProfileExperienceAddController.$inject = ['$scope', 'profileStateService', 'usersService', 'exception'];
 
  /* @ngInject */
  function ProfileExperienceAddController($scope, profileStateService, usersService, exception) {
 
    var vm = this;
 
    vm.addingExperience = false;
    vm.newExperience = {
      Skills: [],
      ProjectLinks: ['']
    };
    vm.newExperienceDateRange = {
      Start: null,
      End: null
    };
    vm.addExperienceErrors = [];
    vm.newExperienceNewSkill = '';
    vm.addSkill = addSkill;
    vm.removeSkill = removeSkill;
    vm.addProjectLink = addProjectLink;
    vm.removeProjectLink = removeProjectLink;
    vm.addNewExperience = addNewExperience;
 
    vm.init = init;
 
    init();
 
    function init() {
      $scope.$watch('vm.newExperienceDateRange.Start', function () {
 
        if (vm.newExperienceDateRange.Start != null) {
          vm.newExperience.StartMonth = vm.newExperienceDateRange.Start.getMonth() + 1;
          vm.newExperience.StartYear = vm.newExperienceDateRange.Start.getFullYear();
        }
      });
 
      $scope.$watch('vm.newExperienceDateRange.End', function () {
 
        if (vm.newExperienceDateRange.End != null) {
          vm.newExperience.EndMonth = vm.newExperienceDateRange.End.getMonth() + 1;
          vm.newExperience.EndYear = vm.newExperienceDateRange.End.getFullYear();
        }
      });
    }
 
 
    /**adds a new skill to the new experience event */
    function addSkill() {
 
      var skillExist = vm.newExperience.Skills.filter(function (s) {
        return s == vm.newExperienceNewSkill;
      }).length > 0;
 
      if (!skillExist) {
        vm.newExperienceEditSkills = false;
        vm.newExperience.Skills.push(vm.newExperienceNewSkill);
      }
      vm.newExperienceNewSkill = '';
    };
 
    /**removes a skill from the new experience event */
    function removeSkill(indexToRemove) {
 
      vm.newExperience.Skills.splice(indexToRemove, 1);
    };
 
    /**adds a new project link from the new experience event  */
    function addProjectLink() {
 
      if (vm.newExperience.ProjectLinks.indexOf("") == -1) {
        vm.newExperience.ProjectLinks.push("");
      }
    };
 
    /**removes a project linkn from the new experience event */
    function removeProjectLink(indexToRemove) {
 
      vm.newExperience.ProjectLinks.splice(indexToRemove, 1);
    };
 
    /**ads the new experience event */
    function addNewExperience() {
 
      //TODO fix this accordingly to newEventAdded in the parent
      if (profileStateService.currentUser.ExperienceEvents.length == 0) {
        $scope.hasFirstEventAdded = profileStateService.currentUser.ExperienceEvents.length == 0;
      }
 
      return usersService.addExperience(vm.newExperience)
        .then(success, error);
 
      function success(data) {
 
        profileStateService.currentUser.ExperienceEvents.push(data);
 
        vm.addingExperience = false;
        vm.newExperienceAddingProjectLinks = false;
        vm.newExperience = {
          Skills: [],
          ProjectLinks: [""]
        };
        vm.addExperienceErrors = [];
        vm.newExperienceDateRange = {
          Start: null,
          End: null
        };
      };
 
      function error(errors) {
        vm.addExperienceErrors = errors;
      }
    };
  }
})();
