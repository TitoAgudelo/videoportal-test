(function () {
  'use strict';
 
  angular
    .module('app.profile')
    .controller('ProfileExperienceEditController', ProfileExperienceEditController);
 
  ProfileExperienceEditController.$inject = ['$scope', 'usersService', 'profileStateService', '$timeout'];
 
  /* @ngInject */
  function ProfileExperienceEditController($scope, usersService, profileStateService, $timeout) {
 
    var vm = this;
 
    vm.experience = {};
    vm.editDateRange = {
      Start: null,
      End: null
    };
    vm.editAddSkill = {
      newSkill: "",
      adding: false
    };
    vm.addSkill = addSkill;
    vm.removeSkill = removeSkill;
    vm.addProjectLink = addProjectLink;
    vm.removeProjectLink = removeProjectLink;
    vm.exitEditMode = exitEditMode;
    vm.save = save;
    vm.remove = remove;
    vm.errors = [];
    vm.init = init;
 
    /**initializes the controller with the single experience event instance this controller controls */
    function init(experience) {
      vm.experience = experience;
 
      if (experience.StartYear) {
        vm.editDateRange.Start = new Date(experience.StartYear, experience.StartMonth - 1);
      }
      if (experience.EndYear) {
        vm.editDateRange.End = new Date(experience.EndYear, experience.EndMonth - 1);
      }
 
      $scope.$watch("vm.editDateRange.Start", function () {
        if (vm.editDateRange.Start != null && vm.experience != null) {
          vm.experience.StartMonth = vm.editDateRange.Start.getMonth() + 1;
          vm.experience.StartYear = vm.editDateRange.Start.getFullYear();
        }
      });
 
      $scope.$watch("vm.editDateRange.End", function () {
        if (vm.editDateRange.End != null && vm.experience != null) {
          vm.experience.EndMonth = vm.editDateRange.End.getMonth() + 1;
          vm.experience.EndYear = vm.editDateRange.End.getFullYear();
        }
      });
    }
 
 
    function exitEditMode() {
      $scope.$emit('profile.experience.edit.exitEditMode');
    }
 
 
    function addSkill() {
      vm.experience.Skills.push(vm.editAddSkill.newSkill);
      vm.editAddSkill.adding = false;
      vm.editAddSkill.newSkill = '';
    }
 
    function removeSkill(indexToRemove) {
      vm.experience.Skills.splice(indexToRemove, 1);
    }
 
    function addProjectLink() {
      if (vm.experience.ProjectLinks.indexOf("") == -1) {
        vm.experience.ProjectLinks.push("");
      }
    }
 
    function removeProjectLink(indexToRemove) {
      vm.experience.ProjectLinks.splice(indexToRemove, 1);
    }
 
    function save() {
 
      usersService.editExperience(vm.experience)
        .then(success, error);
 
      function success() {
 
        vm.exitEditMode();
 
        $timeout(function () {
          usersService.getExperience(vm.experience.ResumeElementId)
            .then(function (data) {
 
              var idx = profileStateService.currentUser.ExperienceEvents.indexOf(vm.experience);
              profileStateService.currentUser.ExperienceEvents[idx] = data;
 
            });
 
        }, 2500);
      }
 
      function error(errors) {
        vm.errors = errors;
      }
 
    }
 
    function remove() {
 
      usersService.removeExperience(vm.experience.ResumeElementId)
        .then(success, error);
 
      function success() {
 
        profileStateService.currentUser.ExperienceEvents
          .splice(profileStateService.currentUser.ExperienceEvents.indexOf(vm.experience), 1);
 
        vm.exitEditMode();
 
        $timeout(function () {
 
          // load user scores
 
        }, 2500);
      }
 
      function error(errors) {
        vm.errors = errors;
      }
    }
 
  }
})();
