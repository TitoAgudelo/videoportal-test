(function () {
  'use strict';
 
  angular
    .module('app.account')
    .controller('SignupController', SignupController);
 
  SignupController.$inject = ['logger', '$scope', '$rootScope', '$http', '$uibModalInstance', '$timeout', '$interval', 'action', 'siteSettings', '$state', 'authService'];
 
  /* @ngInject */
  function SignupController(logger, $scope, $rootScope, $http, $uibModalInstance, $timeout, $interval, action, siteSettings, $state, authService) {
    var vm = this;
    vm.title = 'Signup';
 
 
    $scope.resumePending = null;
    $scope.showScanning = false;
    $scope.scanMessage1 = false;
    $scope.scanMessage2 = false;
    $scope.showForm = false;
    $scope.showSuccess = false;
    $scope.progressBar = false;
    $scope.almostThere = false;
    $scope.joinNow = false;
 
    $scope.user = {
      name: '',
      username: '',
      email: '',
      password: ''
    };
 
 
    if (action === 'upload') {
      $scope.resumePending = true;
      $scope.showScanning = true;
      $scope.scanMessage1 = true;
 
      $scope.dynamic = 0;
 
      $interval(function () {
 
        if ($scope.resumePending == true) {
 
          if ($scope.dynamic == 98) {
            //do nothing
            //this prevents loader from completing unless resume parses successfully
          }
          else if ($scope.dynamic > 50) {
            $scope.dynamic++;
          }
          else {
            $scope.dynamic = $scope.dynamic + 5;
          }
 
          if ($scope.dynamic == 100) {
            $scope.showSuccess = true;
            $scope.almostThere = false;
            $scope.progressBar = false;
            $scope.resumePending = false;
            $scope.showScanning = false;
            $scope.showForm = true;
            $scope.progressBar = true;
          }
          else if ($scope.dynamic > 40) {
            $scope.showScanning = false;
            $scope.almostThere = true;
            $scope.showForm = true;
            $scope.progressBar = true;
          }
          else if ($scope.dynamic > 5) {
            $scope.scanMessage1 = false;
            $scope.scanMessage2 = true;
          }
        }
      }, 1800);
 
      //checkResumeStatus
      $interval(function () {
        $scope.checkUploadStatus();
      }, 1000);
 
    } else if (action === 'join') {
      $scope.showScanning = false;
      $scope.showForm = true;
      $scope.progressBar = false;
      $scope.joinNow = true;
    }
 
    $scope.checkUploadStatus = function () {
 
      if ($scope.resumePending === false) return;
 
      var postData = {
        resumeUploadOperationId: $rootScope.uploadOperationId
      };
 
      $http({
        method: 'GET',
        url: siteSettings.apiBaseUrl + '/api/v4/resume-upload-operations/' + $rootScope.uploadOperationId + '/status',
        data: postData
      }).success(function (data, status) {
 
        if (data.Success === true && data.Status === "Done") { //the resume parsed successfully 
          $scope.dynamic = 100;
 
          //request data parsed from resume
          $http({
            method: 'GET',
            url: siteSettings.apiBaseUrl + '/api/v4/users/me/resume-upload-operation/' + $rootScope.uploadOperationId
          }).success(function (data, status) {
 
            if (data.Success === true) {
              if ($scope.user.name == "") {
                $scope.user.name = data.UserDetails.FirstName + " " + data.UserDetails.LastName;
              }
              if ($scope.user.email == "") {
                $scope.user.email = data.UserDetails.Email;
              }
            }
 
          });
        }
        else if (data.Success === true && data.Status === "Failed") {
 
          $scope.resumePending = false;
        }
 
      });
    }
 
    $scope.checkUsername = function () {
 
      if (!$scope.user.username || $scope.user.username.trim() == "") return;
 
      $http({
        method: 'GET',
        url: siteSettings.apiBaseUrl + '/account/UsernameIsValid',
        params: {
          username: $scope.user.username
        }
      }).success(function (data, status) {
 
        if (data === true) {
          $scope.signup.username.$setValidity("usernameAvailable", true);
        }
        else {
          $scope.signup.username.$setValidity("usernameAvailable", false);
        }
 
      });
 
    };
 
    $scope.submit = function () {
 
      var postData = {
        Name: $scope.user.name,
        Email: $scope.user.email,
        Username: $scope.user.username,
        Password: $scope.user.password
      };
 
      if ($rootScope.uploadOperationId) {
        postData.ResumeUploadOperationId = $rootScope.uploadOperationId;
      }
 
      authService.signup(postData)
        .then(signupSuccess, signupFail);
 
      function signupSuccess(response) {
        $state.go('onboarding');
      }
 
      function signupFail(err) {
        if (data.Errors) {
          for (key in data.Errors) {
            $scope.signupErrorMsg = key + ": " + data.Errors[key];
            break;
          }
        }
        else {
          $scope.signupErrorMsg = "Sorry, there was an error signing you up.";
        }
      }
    }
 
  }
})();
