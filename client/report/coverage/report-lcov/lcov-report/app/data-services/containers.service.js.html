    function search(id, containers) {
        for (var i = 0; i < containers.length; i++) {
            if (containers[i].ContainerId === id) {
                return containers[i];
            }
        }
    }
 
    function update(newContainer, containers) {
        for (var i = 0; i < containers.length; i++) {
            if (containers[i].ContainerId === newContainer.ContainerId) {
                containers[i].ContainerId = newContainer.ContainerId
                containers[i].ContainerName = newContainer.ContainerName;
                containers[i].ContainerDescription = newContainer.ContainerDescription;
                containers[i].ContainerFollowingCount = newContainer.ContainerFollowingCount;
                containers[i].ContainerColor = newContainer.ContainerColor;
                containers[i].ContainerPicture = newContainer.ContainerPicture;
            }
        }
        return containers;
    }
 
    function deleteContainer(id, containers) {
        for (var i = 0; i < containers.length; i++) {
            if (containers[i].ContainerId === id) {
                containers.splice(i, 1);
            }
        }
        return containers;
    }
 
    function guid() {
        function s4() {
            return Math.floor((1 + Math.random()) * 0x10000)
                  .toString(16)
                  .substring(1);
        }
        return s4() + s4() + '-' + s4() + '-' + s4() + '-' +
              s4() + '-' + s4() + s4() + s4();
    }
 
    function getCommunityContainers(userName) {
        if (useTestData && endPoint) {
            _community.containers = test.users_me_containers;
            return _community.containers;

        return $http.get(siteSettings.apiBaseUrl + '/api/v4/community/users/me/containers')
            .then(getContainersComplete)
            .catch(getContainersFailed);
 
        function getContainersComplete(response) {
            return response.data;

        }

    function getContainersByUsername(username) {
        return $http.get(siteSettings.apiBaseUrl + '/api/v4/community/users/'+ username +'/containers')
            .then(getContainersComplete)
            .catch(getContainersFailed);
 
        function getContainersComplete(response) {
            return response.data;
        }
 
        function getContainersFailed(error) {
            $state.go('404');
        }
    }
 
    function getCommunityContainer(userName, containerId) {
        if (useTestData && endPoint) {
            _community.container = search(containerId, _community.containers);
            return _community.container;
        }
        return $http.get(siteSettings.apiBaseUrl + '/api/v4/community/users/'+ userName +'/containers/' + containerId)
            .then(getContainersComplete)
            .catch(getContainersFailed);
 
        function getContainersComplete(response) {
            return response.data;
        }
 
        function getContainersFailed(error) {
            logger.error('XHR Failed for getContainers.' + error.data);
        }
    };
 
    function createCommunityContainer(newContainer) {
        var id = guid();
        var self = this;
        newContainer.ContainerId = "" + id;
        newContainer.ContainerFollowingCount = 0;
 
        if (useTestData && endPoint) {
            _community.containers.push(newContainer);
            return _community.containers;
        }
        var promise = $http({
            method: 'PUT',
            url: siteSettings.apiBaseUrl + '/api/v4/community/users/me/containers',
            data: newContainer
        });
 
        return promise.success(function (data, status) {
            if (data) {
                return data;
            }
        });
    };

        if (useTestData && endPoint) {
            _community.containers = deleteContainer(containerId, _community.containers);
            return _community.containers;
        }
        return $http.delete(siteSettings.apiBaseUrl + '/api/v4/community/users/me/containers/' + containerId)
            .then(getContainersComplete)
            .catch(getContainersFailed);
 
        function getContainersComplete(response) {
            return response;
        }
 
        function getContainersFailed(error) {
            logger.error('XHR Failed for getContainers.' + error.data);
        }
    };
 
    function updateCommunityContainer(newContainer) {
        if (useTestData && endPoint) {
            _community.containers = update(newContainer, _community.containers);
            return _community.containers;
        }
        var id = newContainer.ContainerId;
        var promise = $http({
            method: 'POST',
            url: siteSettings.apiBaseUrl + '/api/v4/community/users/me/containers/' + id,
            data: newContainer
        });
        return promise.success(function (data, status) {
            if (data) {
                return data;
            }
        });
    };
 
    function upload(file) {
        if (file) {
            var promise = Upload.upload({

                return data;
