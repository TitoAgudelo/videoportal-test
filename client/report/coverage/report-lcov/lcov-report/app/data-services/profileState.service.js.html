(function () {
  'use strict';
 
  angular
    .module('app.data')
    .factory('profileStateService', profileStateService);
 
  profileStateService.$inject = ['$http', '$rootScope', '$q', 'exception', 'logger', '$timeout', 'siteSettings', 'usersService'];
 
  /* @ngInject */
  function profileStateService($http, $rootScope, $q, exception, logger, $timeout, siteSettings, usersService) {
 
    var service = {
      currentUser: null,
      activeBucket: '',
      hasActiveBucket: hasActiveBucket,
      loadCurrentUserDetails: loadCurrentUserDetails,
      startLoadingCurrentUserScore: startLoadingCurrentUserScore,
      currentUserHasScore: currentUserHasScore
    };
 
    return service;
 
    /**determines if the current user has scores */
    function currentUserHasScore() {
      if (service.currentUser == null) {
        return false;
      }
      else {
        return service.currentUser.Scores.TotalScore > 0;
      }
    }
 
    /**loads details about the current logged in user, keep state in `currentUser` property */
    function loadCurrentUserDetails(validate) {
      return usersService.getUserDetails('me', validate)
        .then(function (data) {
          service.currentUser = data;
          $rootScope.username = data.Username;
          service.activeBucket = data.DesiredIndexingBucket;
          return data;
        });
    }
 
    /**determines if the activeBucket has a value */
    function hasActiveBucket() {
      return service.activeBucket && service.activeBucket.length > 0;
    }
 
 
    /**loads current logged in user scores from the server and updates the state.
    will load scores every 1 second until some score is returned. 
    */
    function startLoadingCurrentUserScore() {
 
      return usersService.getCurrentUserScores()
        .then(loadAgainIfZeroScore);
 
      function loadAgainIfZeroScore(data) {
        if (data.TotalScore == 0) {
          $timeout(function () {
            service.startLoadingCurrentUserScore();
          }, 1000);
        }
        else {
          service.currentUser.Scores = data;
        }
      }
    }
 
  }
})();
