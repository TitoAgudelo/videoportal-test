    function locationFilter(locationObj) {
 
      if (locationObj.City || locationObj.State || locationObj.Country) {
 
        locationObj.RawLocation = "";
 
        if (locationObj.City) {
          locationObj.RawLocation += locationObj.City;
        }
 
        if (locationObj.State) {
          if (locationObj.RawLocation.length > 0) {
            locationObj.RawLocation += ", ";
          }
          locationObj.RawLocation += locationObj.State;
        }
 
        if (locationObj.Country) {

          }
          locationObj.RawLocation += locationObj.Country;
        }
      }
