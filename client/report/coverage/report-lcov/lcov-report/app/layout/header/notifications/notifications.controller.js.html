(function () {
  'use strict';
 
  angular
    .module('app.layout')
    .controller('NotificationsController', NotificationsController);
 
  NotificationsController.$inject = ['$q', '$rootScope', '$scope', '$timeout', 'config', 'logger',
    'usersService', 'profileStateService', 'notificationsStateService', 'authService', '$location'];
 
  /* @ngInject */
  function NotificationsController($q, $rootScope, $scope, $timeout, config, logger, usersService, profileStateService,
    notificationsStateService, authService, $location) {
 
    var continuousLoadInterval = 1;
 
    var vm = this;
 
    vm.notificationsState = notificationsStateService;
    vm.markNotificationAsRead = markNotificationAsRead;
    vm.ignoreNotification = ignoreNotification;
 
    $scope.markNotificationAsRead = markNotificationAsRead;
    $scope.ignoreNotification = ignoreNotification;
 
    activate();
 
    function activate() {
 
      continuouslyLoadNotifications()
 
      // var activationPromises = [
      //   notificationsStateService.loadNotifications()
      // ];
      //return $q.all(activationPromises);
 
    }
 
    /*marks a notification as read */
    function markNotificationAsRead(notificationIndex, notificationId) {
 
      window.event.stopPropagation();
 
      return notificationsStateService.markNotificationAsRead(notificationId);
    };
 
    /*marks a notification as ignored */
    function ignoreNotification(notificationIndex, notificationId) {
 
      window.event.stopPropagation();
 
      return notificationsStateService.ignoreNotification(notificationId);
    };
 
    /**loads notifications from the server continuously every certain time interval */
    function continuouslyLoadNotifications() {
 
      $timeout(function () {
 
        if (authService.isAuthenticated == true) {
          continuousLoadInterval += 2.5;
          notificationsStateService.loadNotifications();
        }
 
        continuouslyLoadNotifications();
 
      }, continuousLoadInterval * 1000);
    }
  }
})();
