(function () {
  'use strict';
 
  angular
    .module('app.layout')
    .controller('ResumeUploadStatusNotificationController', ResumeUploadStatusNotificationController);
 
  ResumeUploadStatusNotificationController.$inject = ['$q', '$rootScope', '$scope', '$timeout', 'usersService', 'profileStateService', 'notificationsStateService', 'resumeUploadService'];
 
  /* @ngInject */
  function ResumeUploadStatusNotificationController($q, $rootScope, $scope, $timeout, usersService, profileStateService, notificationsStateService, resumeUploadService) {
 
    var vm = this;
 
    vm.notification = {};
    vm.init = init;
 
    function init(notification) {
      vm.notification = notification;
 
      loadResumeStatusData();
    }
 
    /**loads resume status based */
    function loadResumeStatusData() {
 
      var resumeUploadOperationId = vm.notification.CustomData.ResumeUploadOperationId;
 
      return resumeUploadService.getResumeStatusByOperationId(resumeUploadOperationId)
        .then(success);
 
      function success(data) {
        vm.notification.CustomData.ResumeStatus = data.Status;
 
        // if status is Pending, get them again, 
        // (it will stop when it changes from pending to failed or somehting else)
        if (data.Status == 'Pending') {
          $timeout(function () {
            loadResumeStatusData();
          }, 1000);
        }
      }
    }
  }
})();
